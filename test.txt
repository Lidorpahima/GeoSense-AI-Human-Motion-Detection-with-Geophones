class HybridGeoModel(nn.Module):
    """
    מודל היברידי המבוסס על EfficientNet-B0 כ-extractor לתמונות,
    ו-MLP לתכונות הסטטיסטיות, ואז איחוד (concat) ו-classifier סופי.
    """
    def __init__(self, dropout_p=0.4, stat_feature_dim=64, combined_dim=128):
        super(HybridGeoModel, self).__init__()

        # טוענים את EfficientNet-B0
        weights = EfficientNet_B0_Weights.DEFAULT
        self.cnn_backbone = efficientnet_b0(weights=weights)

        # --- התאמה ---
        # הדרך הבטוחה: קבל את מספר תכונות הכניסה מהשכבה הלינארית ב-classifier המקורי
        # ב-EfficientNet-B0, ה-classifier הוא nn.Sequential(Dropout, Linear)
        # לכן ניגש לשכבה הלינארית (אינדקס 1) וניקח את in_features
        original_classifier = self.cnn_backbone.classifier
        if isinstance(original_classifier, nn.Sequential) and len(original_classifier) > 1 and isinstance(original_classifier[1], nn.Linear):
            out_channels = original_classifier[1].in_features
        elif isinstance(original_classifier, nn.Linear): # אם במקרה זה רק שכבה לינארית
             out_channels = original_classifier.in_features
        else:
             # Fallback - ייתכן שצריך להתאים אם המבנה שונה מאוד
             # ננסה דרך אחרת או נקבע ערך ידני אם ידוע
             print("Warning: Could not automatically determine out_channels from classifier. Trying features.")
             # נחזור לגישה הקודמת, למקרה שהיא עובדת בגרסה מסוימת
             try:
                 last_feature_block = self.cnn_backbone.features[-1]
                 if hasattr(last_feature_block, 'out_channels'):
                     out_channels = last_feature_block.out_channels
                 elif isinstance(last_feature_block, nn.Sequential) and hasattr(last_feature_block[-1], 'out_channels'):
                      out_channels = last_feature_block[-1].out_channels
                 elif isinstance(last_feature_block, nn.Conv2d):
                      out_channels = last_feature_block.out_channels
                 else: # ננסה לגשת לרכיב הראשון אם זה Conv2dNormActivation
                     out_channels = last_feature_block[0].out_channels

             except Exception as e:
                 print(f"Error in fallback method: {e}. Setting default out_channels=1280 for EfficientNet-B0.")
                 out_channels = 1280 # ערך ידוע עבור EfficientNet-B0

        print(f"[DEBUG] Determined CNN output features: {out_channels}")

        # עכשיו אפשר לבטל את ה-classifier המקורי בבטחה
        self.cnn_backbone.classifier = nn.Identity()
        # -------------

        # MLP לתכונות הסטטיסטיות (ודא ש-NUM_STAT_FEATURES מעודכן למעלה)
        self.stat_mlp = nn.Sequential(
            nn.LayerNorm(NUM_STAT_FEATURES), # משתמש בגודל המעודכן
            nn.Linear(NUM_STAT_FEATURES, stat_feature_dim * 2),
            nn.ReLU(),
            nn.Dropout(dropout_p),
            nn.Linear(stat_feature_dim * 2, stat_feature_dim),
            nn.ReLU()
        )

        # שכבות איחוד
        self.dropout = nn.Dropout(dropout_p)
        # גודל הכניסה ל-fc_combined צריך להיות סכום התכונות מה-CNN וה-MLP
        self.fc_combined = nn.Linear(out_channels + stat_feature_dim, combined_dim)
        self.fc_output = nn.Linear(combined_dim, NUM_CLASSES)

    def forward(self, spec_input, stat_input):
        # 1) CNN על התמונה (spec_input)
        cnn_features = self.cnn_backbone(spec_input) # הפלט הוא כבר אחרי ה-pooling ב-EfficientNet
        # הפלט אחרי nn.Identity() יהיה בגודל (batch_size, out_channels)
        # cnn_features = cnn_features.view(cnn_features.size(0), -1) # לא תמיד נחוץ אם הפלט כבר שטוח
        # 2) MLP על התכונות הסטטיסטיות
        stat_features = self.stat_mlp(stat_input)
        # 3) איחוד
        combined = torch.cat((cnn_features, stat_features), dim=1)
        x = self.dropout(combined)
        x = F.relu(self.fc_combined(x))
        x = self.dropout(x)
        out = self.fc_output(x)
        return out
