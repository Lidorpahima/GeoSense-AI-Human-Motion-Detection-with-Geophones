for fold, (train_idx, val_idx) in enumerate(gkf.split(orig_specs, orig_labels, groups=orig_groups)):
        print(f"\n  >>> Fold {fold+1}/{N_SPLITS} <<<")

        # נתונים מקוריים ל-Fold הנוכחי
        X_spec_train_orig = orig_specs[train_idx]
        X_stat_train_orig = orig_stat_feats[train_idx]
        y_train_orig      = orig_labels[train_idx]

        X_spec_val   = orig_specs[val_idx]
        X_stat_val   = orig_stat_feats[val_idx]
        y_val        = orig_labels[val_idx]

        # החלת SMOTE רק על התכונות הסטטיסטיות והתוויות
        print(f"    Original train samples: {len(y_train_orig)}")
        smote = SMOTE(random_state=42, k_neighbors=min(5, len(y_train_orig)-1)) # התאמת k_neighbors
        X_stat_train_smote, y_train_smote = smote.fit_resample(X_stat_train_orig, y_train_orig)
        print(f"    Train samples after SMOTE: {len(y_train_smote)}")

        # --- התיקון ---
        # יצירת ספקטרוגרמות תואמות לגודל החדש
        num_original_train = len(X_spec_train_orig)
        num_total_smote = len(y_train_smote)

        if num_total_smote > num_original_train:
            # צריך להוסיף ספקטרוגרמות
            num_to_add = num_total_smote - num_original_train
            # דוגמים אינדקסים רנדומליים *מהמקור* כדי לשכפל ספקטרוגרמות
            random_indices = np.random.choice(num_original_train, size=num_to_add, replace=True)
            specs_to_add = X_spec_train_orig[random_indices]
            # מחברים את הספקטרוגרמות המקוריות עם המשוכפלות
            X_spec_train_final = np.concatenate((X_spec_train_orig, specs_to_add), axis=0)
        else:
             # אם SMOTE לא הוסיף דגימות (נדיר, אולי אם המחלקות כבר מאוזנות)
            X_spec_train_final = X_spec_train_orig

        # ודא שכל המערכים באותו גודל
        assert len(X_spec_train_final) == len(X_stat_train_smote) == len(y_train_smote), \
               f"Mismatch after SMOTE! Specs: {len(X_spec_train_final)}, Stats: {len(X_stat_train_smote)}, Labels: {len(y_train_smote)}"
        # -------------

        # Scaling רק *אחרי* SMOTE (על הנתונים המורחבים)
        scaler = StandardScaler()
        X_stat_train_scaled = scaler.fit_transform(X_stat_train_smote) # שימוש בנתונים אחרי SMOTE
        X_stat_val_scaled   = scaler.transform(X_stat_val) # שימוש בנתוני ה-Val המקוריים

        # יצירת ה-Datasets עם הגדלים התואמים
        train_dataset = HybridDataset(X_spec_train_final, X_stat_train_scaled, y_train_smote, transform=current_train_transforms)
        val_dataset   = HybridDataset(X_spec_val, X_stat_val_scaled, y_val, transform=val_transforms)

        # המשך הקוד כרגיל...
        train_loader = DataLoader(train_dataset, batch_size=BATCH_SIZE, shuffle=True, num_workers=8)
        val_loader   = DataLoader(val_dataset,   batch_size=BATCH_SIZE, shuffle=False, num_workers=8)
