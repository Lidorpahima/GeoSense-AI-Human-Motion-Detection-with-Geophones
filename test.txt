# (pad_channels, create_windows, create_mel_spectrogram - ללא שינוי מהותי)
def pad_channels(data, target_channels=max_channels):
    if data.ndim == 1: data = data.reshape(-1, 1); current_channels = 1
    else: current_channels = data.shape[1]
    if current_channels == target_channels: return data
    elif current_channels < target_channels: padding = np.zeros((data.shape[0], target_channels - current_channels), dtype=data.dtype); return np.hstack((data, padding))
    else: return data[:, :target_channels]
def create_windows(signal, label, window_size=window_size, stride=stride):
    X_windows, y_labels = [], []
    if signal.shape[0] < window_size: return [], []
    for start in range(0, signal.shape[0] - window_size + 1, stride):
        X_windows.append(signal[start : start + window_size, :]); y_labels.append(label)
    return X_windows, y_labels
def create_mel_spectrogram(window_1d, sr=int(fs), n_fft=N_FFT, hop_length=HOP_LENGTH, n_mels=N_MELS, fmin=lowcut, fmax=highcut):
    try:
        signal = window_1d.astype(np.float32);
        if np.all(signal == signal[0]): return None
        effective_fmax = min(fmax, sr / 2 - 1);
        if effective_fmax <= fmin: effective_fmax = fmin + 10
        mel_spec = librosa.feature.melspectrogram(y=signal, sr=int(sr), n_fft=int(n_fft), hop_length=int(hop_length), n_mels=int(n_mels), fmin=fmin, fmax=effective_fmax)
        mel_spec_db = librosa.power_to_db(mel_spec, ref=np.max); min_db, max_db = mel_spec_db.min(), mel_spec_db.max()
        if max_db <= min_db: return None
        mel_spec_norm = (mel_spec_db - min_db) / (max_db - min_db) if max_db > min_db else np.zeros_like(mel_spec_db)
        if not np.isfinite(mel_spec_norm).all(): return None
        return mel_spec_norm.astype(np.float32)
    except Exception as e: print(f"  Error in create_mel_spectrogram: {e}"); return None

# --- פונקציית טעינה משופרת (עם תיקון המרה ל-float32) ---
def load_and_process_data_grouped(config_map, target_img_size=(IMG_HEIGHT_FINAL, IMG_WIDTH_FINAL)):
    all_specs, all_labels, all_groups = [], [], []
    try: b, a = butter(order, [lowcut/(fs/2), highcut/(fs/2)], btype='band')
    except ValueError as e: print(f"CRITICAL FILTER ERROR: {e}. Aborting."); return [], [], []
    print("Processing data sources...")
    total_items_created = 0
    for source_idx, config in config_map.items():
        filepath = config.get('path'); label = config.get('label'); group_id = config.get('group_id'); extra_args = config.get('args', {})
        if not filepath or label is None or group_id is None: print(f"Skipping source {source_idx}: Missing info."); continue
        if not os.path.exists(filepath): print(f"Warning: File not found for source {source_idx}: {filepath}. Skipping."); continue
        print(f"\nProcessing Source {source_idx}: File={os.path.basename(filepath)}, Label={label}, Group={group_id}")
        df = None; data = None; windows_created_count = 0; window_processed_count = 0
        try:
            try: df = pd.read_csv(filepath, header=None, skipinitialspace=True, **extra_args)
            except Exception: df = pd.read_csv(filepath, header=None, skiprows=1, skipinitialspace=True, **extra_args)
            if df is None or df.empty: print("  Empty file/read failed. Skipping."); continue
            df = df.dropna();
            if df.empty: print("  Empty after dropna. Skipping."); continue
            # <<< תיקון: נסה להמיר ישירות ל-float32 >>>
            try: data = df.values.astype(np.float32)
            except ValueError as e_type: print(f"  ERROR converting to float32: {e_type}. Skipping."); continue
            if data.shape[0] < window_size or data.ndim < 2 or data.shape[1] == 0: print(f"  Insufficient data shape {data.shape}. Skipping."); continue
            data_padded = pad_channels(data);
            try: filtered_data = filtfilt(b, a, data_padded, axis=0)
            except Exception as e_filter: print(f"  ERROR filtering data: {e_filter}. Skipping."); continue
            windows, labels_w = create_windows(filtered_data, label)
            if not windows: print("  No windows created."); continue
            windows_created_count = len(windows)
            for i, w in enumerate(windows):
                try:
                    signal_1d = w[:, 0] # השתמש בערוץ הראשון
                    mel_spec_norm = create_mel_spectrogram(signal_1d)
                    if mel_spec_norm is None: continue
                    resized_spec = cv2.resize(mel_spec_norm, (target_img_size[1], target_img_size[0]), interpolation=cv2.INTER_LINEAR)
                    spec_3_channels = np.stack([resized_spec]*3, axis=0)
                    if not np.isfinite(spec_3_channels).all(): continue
                    all_specs.append(spec_3_channels); all_labels.append(labels_w[i]); all_groups.append(group_id)
                    window_processed_count += 1; total_items_created += 1
                except Exception: pass
            print(f"  -> Processed {window_processed_count}/{windows_created_count} windows.")
        except Exception as e_file: print(f"  UNEXPECTED ERROR processing source {source_idx}: {e_file}")
    print(f"\nFinished processing all sources. Total items created: {total_items_created}")
    if not all_specs: print("WARNING: No spectrograms created.")
    specs_np = np.array(all_specs, dtype=np.float32) if all_specs else np.array([])
    labels_np = np.array(all_labels, dtype=np.int64) if all_labels else np.array([], dtype=np.int64)
    groups_np = np.array(all_groups, dtype=np.int64) if all_groups else np.array([], dtype=np.int64)
    del all_specs, all_labels, all_groups; return specs_np, labels_np, groups_np

# (הגדרות Dataset, Augmentation, train/evaluate functions - דומות לקודם)
class SpecAugment(nn.Module):
    def __init__(self, freq_mask_param=20, time_mask_param=40, num_freq_masks=1, num_time_masks=1):
        super().__init__(); self.freq_mask_param = freq_mask_param; self.time_mask_param = time_mask_param; self.num_freq_masks = num_freq_masks; self.num_time_masks = num_time_masks
    def forward(self, spec):
        if not isinstance(spec, torch.Tensor): raise TypeError("Input must be a Tensor.");
        if spec.ndim != 3: print(f"ERROR in SpecAugment: Expected 3D, got {spec.ndim}D {spec.shape}. Skipping."); return spec
        spec_aug = spec.clone(); C, H, W = spec_aug.shape
        if H <= 0 or W <= 0: print(f"Warning in SpecAugment: Zero size dim ({spec.shape}). Skipping."); return spec
        fm_param = min(self.freq_mask_param, H // 2); tm_param = min(self.time_mask_param, W // 2); mask_value = spec.mean()
        for _ in range(self.num_freq_masks):
            if fm_param > 0: f = random.randrange(0, fm_param + 1);
            if H >= f and f > 0: f0 = random.randrange(0, H - f + 1); spec_aug[:, f0 : f0 + f, :] = mask_value
        for _ in range(self.num_time_masks):
             if tm_param > 0: t = random.randrange(0, tm_param + 1);
             if W >= t and t > 0: t0 = random.randrange(0, W - t + 1); spec_aug[:, :, t0 : t0 + t] = mask_value
        return spec_aug

class SpectrogramDataset(Dataset):
    def __init__(self, specs_np, labels_np, transform=None):
        self.specs = torch.from_numpy(specs_np.astype(np.float32)); self.labels = torch.from_numpy(labels_np.astype(np.int64)); self.transform = transform
    def __len__(self): return len(self.labels)
    def __getitem__(self, idx):
        spec = self.specs[idx]; label = self.labels[idx];
        if self.transform: spec = self.transform(spec)
        return spec, label

def train_epoch(model, dataloader, criterion, optimizer, device):
    model.train(); running_loss = 0.0; correct_preds = 0; total_samples = 0;
    for inputs, labels in dataloader:
        try:
            inputs, labels = inputs.to(device), labels.to(device); optimizer.zero_grad(); outputs = model(inputs); loss = criterion(outputs, labels)
            if loss is None or torch.isnan(loss): continue
            preds = outputs.max(1)[1]; loss.backward(); optimizer.step()
            running_loss += loss.item() * inputs.size(0); correct_preds += (preds == labels).sum().item(); total_samples += labels.size(0)
        except Exception as e: print(f"  Error train batch: {e}"); traceback.print_exc(); continue
    if total_samples == 0: return 0.0, 0.0; return running_loss / total_samples, correct_preds / total_samples

def evaluate_model(model, dataloader, criterion, device):
    model.eval(); running_loss = 0.0; correct_preds = 0; total_samples = 0; all_preds = []; all_labels = []
    with torch.no_grad():
        for inputs, labels in dataloader:
            try:
                inputs, labels = inputs.to(device), labels.to(device); outputs = model(inputs); loss = criterion(outputs, labels);
                if loss is None or torch.isnan(loss): continue
                preds = outputs.max(1)[1]
                running_loss += loss.item() * inputs.size(0); correct_preds += (preds == labels).sum().item(); total_samples += labels.size(0)
                all_preds.extend(preds.cpu().numpy()); all_labels.extend(labels.cpu().numpy())
            except Exception as e: print(f"  Error eval batch: {e}"); traceback.print_exc(); continue
    accuracy = correct_preds / total_samples if total_samples > 0 else 0.0
    avg_loss = running_loss / total_samples if total_samples > 0 else 0.0
    return avg_loss, accuracy, all_labels, all_preds

# --- פונקציית בניית מודל MobileNetV2 ---
def setup_mobilenetv2_model(pretrained=True, num_classes=NUM_CLASSES, dropout_p=0.2):
    # (אותה פונקציה כמו קודם)
    print(f"Setting up MobileNetV2 (Pretrained: {pretrained})...")
    weights = MobileNet_V2_Weights.DEFAULT if pretrained else None
    model_arch = mobilenet_v2(weights=weights)
    try:
        num_ftrs = model_arch.classifier[1].in_features
        model_arch.classifier = nn.Sequential(nn.Dropout(p=dropout_p, inplace=True), nn.Linear(num_ftrs, num_classes))
        print(f"Replaced MobileNetV2 classifier for {num_classes} classes with dropout {dropout_p}.")
    except Exception as e: print(f"Error replacing MobileNetV2 classifier: {e}"); return None
    if pretrained:
        print("Freezing feature parameters...")
        for param in model_arch.features.parameters(): param.requires_grad = False
        for param in model_arch.classifier.parameters(): param.requires_grad = True
        print("Feature parameters frozen.")
    model_arch = model_arch.to(DEVICE); return model_arch

print("\n--- Defining Optuna Objective Function ---")

def objective(trial):
    start_trial_time = time.time()
    print(f"\n{'='*15} Optuna Trial {trial.number} Starting {'='*15}")

    # --- דגימת היפרפרמטרים ---
    lr = trial.suggest_float("lr", 1e-4, 5e-3, log=True)
    dropout_p = trial.suggest_float("dropout_p", 0.1, 0.5, step=0.1) # טווח דרופאאוט למוביילנט
    weight_decay = trial.suggest_float("weight_decay", 1e-5, 1e-3, log=True)
    specaug_freq = trial.suggest_int("specaug_freq", 5, 30) # אפשר להרחיב מעט
    specaug_time = trial.suggest_int("specaug_time", 10, 60)
    specaug_f_masks = trial.suggest_int("specaug_f_masks", 1, 3)
    specaug_t_masks = trial.suggest_int("specaug_t_masks", 1, 3)
    print(f"  Trial Params: LR={lr:.1e}, Dropout={dropout_p:.2f}, WD={weight_decay:.1e}, SpecAugF={specaug_freq}, SpecAugT={specaug_time}, N_F={specaug_f_masks}, N_T={specaug_t_masks}")

    # --- טרנספורמציות אימון ---
    current_train_transforms = transforms.Compose([
        SpecAugment(freq_mask_param=specaug_freq, time_mask_param=specaug_time, num_freq_masks=specaug_f_masks, num_time_masks=specaug_t_masks),
        transforms.Normalize(mean=imagenet_mean, std=imagenet_std) ])

    # --- Cross-Validation על הסט המאוזן ---
    kfold = StratifiedGroupKFold(n_splits=N_SPLITS_OPT, shuffle=True, random_state=trial.number + SEED_OFFSET)
    fold_val_accs = []
    fold_research_accs = [] # לאחסון דיוק על סט המחקר
    qualified_fold_results_in_trial = [] # לאחסון תוצאות מלאות (כולל state_dict) של פולדים שעמדו בתנאי הראשוני

    X, y, groups = train_val_specs_np, train_val_labels_np, train_val_groups_np
    unique_groups_cv = np.unique(groups)
    if len(unique_groups_cv) < N_SPLITS_OPT: print(f"  WARNING: Unique groups for CV ({len(unique_groups_cv)}) < N_SPLITS_OPT ({N_SPLITS_OPT}).")
    try: fold_iterator = kfold.split(X, y, groups)
    except ValueError as e_kfold: print(f"  ERROR during kfold.split: {e_kfold}. Pruning."); raise optuna.exceptions.TrialPruned(f"KFold split error: {e_kfold}")

    # --- לולאת ה-Folds ---
    for fold, (train_ids, val_ids) in enumerate(fold_iterator):
        fold_start_time = time.time(); print(f"  --- Fold {fold+1}/{N_SPLITS_OPT} ---")
        train_groups, val_groups = np.unique(groups[train_ids]), np.unique(groups[val_ids]); overlap = len(set(train_groups) & set(val_groups)); print(f"    Train groups: {len(train_groups)}, Val groups: {len(val_groups)}, Overlap: {overlap}");
        if overlap > 0: print("    !!!! CRITICAL WARNING: Group overlap detected. Leakage occurring! !!!!")

        X_train_fold, y_train_fold = X[train_ids], y[train_ids]; X_val_fold, y_val_fold = X[val_ids], y[val_ids]
        if len(X_train_fold) == 0 or len(X_val_fold) == 0: print(f"    Warning: Fold {fold+1} empty train/val set. Skipping."); fold_val_accs.append(0.0); fold_research_accs.append(0.0); continue

        train_dataset = SpectrogramDataset(X_train_fold, y_train_fold, transform=current_train_transforms); val_dataset = SpectrogramDataset(X_val_fold, y_val_fold, transform=val_test_transforms)
        use_pin_memory = (DEVICE == torch.device("cuda")); train_loader = DataLoader(train_dataset, batch_size=BATCH_SIZE_OPT, shuffle=True, num_workers=0, pin_memory=use_pin_memory, drop_last=True); val_loader = DataLoader(val_dataset, batch_size=BATCH_SIZE_OPT, shuffle=False, num_workers=0, pin_memory=use_pin_memory)

        # --- הגדרת מודל, קריטריון, אופטימייזר ---
        model = setup_mobilenetv2_model(pretrained=True, num_classes=NUM_CLASSES, dropout_p=dropout_p)
        if model is None: print("  ERROR: Failed to create model. Pruning trial."); raise optuna.exceptions.TrialPruned("Model creation failed")
        criterion = nn.CrossEntropyLoss()
        # אופטימיזציה רק לשכבות הלא קפואות (המסווג)
        params_to_optimize = [p for p in model.parameters() if p.requires_grad]
        if not params_to_optimize: print("  ERROR: No trainable parameters found! Check model setup/freezing."); raise optuna.exceptions.TrialPruned("No trainable parameters")
        optimizer = optim.AdamW(params_to_optimize, lr=lr, weight_decay=weight_decay)

        # --- אימון ה-Fold ---
        best_val_acc_fold = 0.0; current_best_fold_model_state = None; epochs_no_improve = 0; best_epoch = -1
        history_current_fold = {'train_loss': [], 'train_acc': [], 'val_loss': [], 'val_acc': []}
        best_fold_val_labels = []; best_fold_val_preds = []
        fold_research_labels = []; fold_research_preds = [] # תוצאות מהערכה על סט מחקר
        current_fold_research_acc = 0.0

        print(f"    Starting training for max {EPOCHS_OPT} epochs...")
        for epoch in range(1, EPOCHS_OPT + 1):
            epoch_start_time = time.time()
            train_loss, train_acc = train_epoch(model, train_loader, criterion, optimizer, device)
            # הערכה על סט הולידציה (המאוזן)
            val_loss, val_acc, fold_val_labels_epoch, fold_val_preds_epoch = evaluate_model(model, val_loader, criterion, device)
            epoch_time = time.time() - epoch_start_time
            print(f"    Epoch {epoch}/{EPOCHS_OPT} | Train Loss: {train_loss:.4f}, Acc: {train_acc:.4f} | Val Loss: {val_loss:.4f}, Acc: {val_acc:.4f} | Time: {epoch_time:.1f}s")
            history_current_fold['train_loss'].append(train_loss); history_current_fold['train_acc'].append(train_acc); history_current_fold['val_loss'].append(val_loss); history_current_fold['val_acc'].append(val_acc)

            if val_acc > best_val_acc_fold:
                best_val_acc_fold = val_acc; current_best_fold_model_state = deepcopy(model.state_dict()); epochs_no_improve = 0; best_epoch = epoch
                best_fold_val_labels = fold_val_labels_epoch; best_fold_val_preds = fold_val_preds_epoch
            else: epochs_no_improve += 1
            if epochs_no_improve >= PATIENCE_OPT: print(f"    Early stopping at epoch {epoch}. Best val_acc: {best_val_acc_fold:.4f} at epoch {best_epoch}."); break

        # --- הערכה על סט המחקר + שמירה מותנית ---
        if current_best_fold_model_state and research_specs_np.size > 0:
            print(f"\n    Fold {fold+1} training finished. Best Val Acc (Epoch {best_epoch}): {best_val_acc_fold:.4f}.")
            # בדוק אם עומד בתנאי הראשוני להערכה על סט המחקר
            if MIN_VAL_ACC_THRESHOLD <= best_val_acc_fold <= MAX_VAL_ACC_THRESHOLD:
                print(f"      Val Acc within evaluation threshold [{MIN_VAL_ACC_THRESHOLD}, {MAX_VAL_ACC_THRESHOLD}]. Evaluating on research data...")
                model.load_state_dict(current_best_fold_model_state) # טען את המצב הטוב ביותר מה-Fold
                research_dataset = SpectrogramDataset(research_specs_np, research_labels_np, transform=val_test_transforms)
                research_loader = DataLoader(research_dataset, batch_size=BATCH_SIZE_OPT, shuffle=False, num_workers=0, pin_memory=use_pin_memory)
                try:
                    # הרץ הערכה על סט המחקר
                    research_loss, research_acc, fold_research_labels_eval, fold_research_preds_eval = evaluate_model(model, research_loader, criterion, device)
                    current_fold_research_acc = research_acc # שמור את הדיוק
                    fold_research_labels = fold_research_labels_eval
                    fold_research_preds = fold_research_preds_eval
                    print(f"        -> Research Data Eval: Loss={research_loss:.4f}, Accuracy={current_fold_research_acc:.4f}")

                    # שמור את כל התוצאות של ה-Fold (כולל state_dict) לשימוש מאוחר יותר ב-Trial
                    qualified_fold_results_in_trial.append({
                        'fold': fold + 1, 'val_acc': best_val_acc_fold, 'research_acc': current_fold_research_acc, # הוסף דיוק מחקר
                        'epoch': best_epoch, 'history': history_current_fold,
                        'val_labels': best_fold_val_labels, 'val_preds': best_fold_val_preds,
                        'research_labels': fold_research_labels, 'research_preds': fold_research_preds, # הוסף תוצאות מחקר
                        'state_dict': deepcopy(current_best_fold_model_state)
                    })

                    # --- בדיקה לשמירה מותנית ---
                    if SAVE_THRESHOLD_VAL_MIN <= best_val_acc_fold <= SAVE_THRESHOLD_VAL_MAX and current_fold_research_acc >= SAVE_THRESHOLD_RESEARCH:
                        print(f"    **** Fold {fold+1} (Trial {trial.number}) meets saving criteria (Val: {best_val_acc_fold:.4f}, Research: {current_fold_research_acc:.4f})! Saving... ****")
                        try:
                            model_filename_cond = os.path.join(QUALIFIED_MODELS_DIR, f"{MODEL_NAME_OPT}_trial_{trial.number}_fold_{fold+1}_val_{best_val_acc_fold:.3f}_res_{current_fold_research_acc:.3f}.pth")
                            save_data_fold = {
                                'model_state_dict': current_best_fold_model_state, 'trial_number': trial.number, 'fold': fold + 1, 'epoch': best_epoch,
                                'val_acc': best_val_acc_fold, 'research_acc': current_fold_research_acc, # שמור את שני הדיוקים
                                'model_name': MODEL_NAME_OPT, 'hyperparameters': trial.params, 'imagenet_mean': imagenet_mean, 'imagenet_std': imagenet_std,
                                'class_names': CLASS_NAMES, 'N_MELS': N_MELS, 'N_FFT': N_FFT, 'HOP_LENGTH': HOP_LENGTH, 'IMG_HEIGHT_FINAL': IMG_HEIGHT_FINAL,
                                'IMG_WIDTH_FINAL': IMG_WIDTH_FINAL, 'window_size': window_size, 'stride': stride, 'fs': fs, 'lowcut': lowcut, 'highcut': highcut, 'order': order,
                                'kfold_random_state': trial.number + SEED_OFFSET,
                                # שמור גם את התוצאות המפורטות (יכול להיות גדול)
                                'history': history_current_fold, 'val_labels': best_fold_val_labels, 'val_preds': best_fold_val_preds,
                                'research_labels': fold_research_labels, 'research_preds': fold_research_preds
                            }
                            torch.save(save_data_fold, model_filename_cond); print(f"      Saved qualified model to: {model_filename_cond}")
                            # אפשר לוותר על קובץ טקסט נפרד כי המידע ב-pth
                        except Exception as e_save_cond: print(f"      ERROR saving qualified model: {e_save_cond}"); traceback.print_exc()
                    # --- סוף שמירה מותנית ---

                except Exception as e_eval_res: print(f"        ERROR during research data evaluation: {e_eval_res}"); traceback.print_exc(); current_fold_research_acc = 0.0
            else: print(f"      Val Acc ({best_val_acc_fold:.4f}) outside evaluation threshold. Skipping research data evaluation.")
        elif research_specs_np.size == 0: print(f"    No research data loaded. Skipping research data evaluation.")
        else: print(f"    No best model state found from training fold {fold+1}. Skipping research data evaluation.")

        fold_val_accs.append(best_val_acc_fold)
        fold_research_accs.append(current_fold_research_acc) # שמור 0 אם לא הוערך

        trial.report(best_val_acc_fold, fold) # דיווח ולידציה ל-Pruning
        if trial.should_prune(): print(f"  Trial {trial.number} pruned at fold {fold+1}."); raise optuna.exceptions.TrialPruned()
        print(f"  --- Fold {fold+1} Finished. Time: {time.time() - fold_start_time:.1f}s ---")
        del model, optimizer, criterion, train_loader, val_loader, train_dataset, val_dataset;
        if DEVICE == torch.device("cuda"): torch.cuda.empty_cache()
    # --- סוף לולאת ה-Folds ---

    # --- סיכום ה-Trial ---
    average_val_acc = np.mean(fold_val_accs) if fold_val_accs else 0.0
    # <<< המטרה למיקסום: ממוצע דיוק על סט המחקר >>>
    qualified_research_accs = [f['research_acc'] for f in qualified_fold_results_in_trial]
    average_research_acc = np.mean(qualified_research_accs) if qualified_research_accs else 0.0

    print(f"\n--- Finished Trial {trial.number} ---")
    print(f"  Avg Val Acc (Balanced Set, all folds): {average_val_acc:.4f}")
    print(f"  Avg Research Acc (Steps Only, qualified folds): {average_research_acc:.4f}")
    print(f"  Number of folds evaluated on research data: {len(qualified_research_accs)}/{N_SPLITS_OPT}")
    print(f"  Total Trial Time: {time.time() - start_trial_time:.1f}s"); print(f"{'='*40}")

    # --- שמירת מידע ב-User Attributes ---
    trial_summary = { 'trial_num': trial.number, 'params': trial.params, 'avg_val_acc': average_val_acc, 'avg_research_acc': average_research_acc,
        'qualified_folds_metrics': [{'fold': f['fold'], 'val_acc': f['val_acc'], 'research_acc': f['research_acc']} for f in qualified_fold_results_in_trial] }
    trial.set_user_attr("trial_summary", trial_summary)
    fold_results_no_state = [{k: v for k, v in fold_res.items() if k != 'state_dict'} for fold_res in qualified_fold_results_in_trial]
    trial.set_user_attr("qualified_fold_results_no_state", fold_results_no_state)

    best_fold_in_trial_state_dict = None; best_fold_num_in_trial = -1
    if qualified_fold_results_in_trial:
        # מיון לפי ולידציה ואז מחקר
        qualified_fold_results_in_trial.sort(key=lambda x: (-x['val_acc'], -x['research_acc']))
        best_qualified_fold_info = qualified_fold_results_in_trial[0]
        best_fold_in_trial_state_dict = best_qualified_fold_info['state_dict']; best_fold_num_in_trial = best_qualified_fold_info['fold']
        print(f"  Identified best fold in trial: Fold {best_fold_num_in_trial} (ValAcc: {best_qualified_fold_info['val_acc']:.4f}, ResearchAcc: {best_qualified_fold_info['research_acc']:.4f})")
        trial.set_user_attr("best_fold_state_dict", best_fold_in_trial_state_dict); trial.set_user_attr("best_fold_num_in_trial", best_fold_num_in_trial);
        print(f"  Stored state_dict for best fold ({best_fold_num_in_trial}) in trial attributes.")
    else: trial.set_user_attr("best_fold_state_dict", None); trial.set_user_attr("best_fold_num_in_trial", -1)

    del qualified_fold_results_in_trial # פינוי זיכרון
    # <<< החזר את מטרת Optuna >>>
    return average_research_acc

                val_metrics={'labels': checkpoint_q.get('val_labels', []), 'preds': checkpoint_q.get('val_preds', [])},
                research_metrics={'labels': checkpoint_q.get('research_labels', []), 'preds': checkpoint_q.get('research_preds', [])},
                class_names=checkpoint_q.get('class_names', CLASS_NAMES), # Use saved names if available
                save_dir=QUALIFIED_MODELS_DIR # שמור גרפים בתיקייה שלהם
            )
        except Exception as e_analyze_q: print(f"  Error analyzing or plotting qualified model {os.path.basename(model_file)}: {e_analyze_q}"); traceback.print_exc()

    # הדפסת סיכום טבלאי בסוף
    if saved_models_summary:
        print("\n--- Summary Table of Qualified Models (Sorted by Research Acc) ---")
        # מיון לפי דיוק מחקר (יורד) ואז ולידציה (יורד)
        saved_models_summary.sort(key=lambda x: (-x['research_acc'], -x['val_acc']))
        print("Filename | Trial | Fold | Epoch | Val Acc | Research Acc | Learning Rate | Dropout")
        print("-------- | ----- | ---- | ----- | ------- | ------------ | ------------- | -------")
        for item in saved_models_summary:
            lr = item['params'].get('lr', 'N/A')
            dropout = item['params'].get('dropout_p', 'N/A')
            lr_str = f"{lr:.1e}" if isinstance(lr, float) else str(lr)
            dropout_str = f"{dropout:.2f}" if isinstance(dropout, float) else str(dropout)
            print(f"{item['filename'][:30]}... | {item['trial']:<5} | {item['fold']:<4} | {item['epoch']:<5} | {item['val_acc']:.4f}  | {item['research_acc']:.4f}       | {lr_str:<11} | {dropout_str:<7}")

print("\n--- Full Script Finished ---")
