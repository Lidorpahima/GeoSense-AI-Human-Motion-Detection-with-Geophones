# --- פונקציית טעינת נתונים מתוקנת ---
def load_data_specs_signals_and_groups(file_map, target_img_size=(IMG_HEIGHT_FINAL, IMG_WIDTH_FINAL)):
    all_specs, all_labels, all_signals, all_groups = [], [], [], []
    try: b, a = butter(order, [lowcut/(fs/2), highcut/(fs/2)], btype='band')
    except ValueError as e: print(f"CRITICAL FILTER ERROR: {e}. Aborting."); return [], [], [], []
    print(f"Loading data, signals, and groups...")
    items_created_count = 0; processed_files = 0; group_counter = 0
    # לולאה על התוויות והמידע במילון הקלט
    for label, info in file_map.items():
        filepaths = info.get('path', []) # קבל את הנתיב/רשימת הנתיבים
        extra_args = info.get('args', {}) # קבל ארגומנטים נוספים (כמו skiprows)

        # ודא ש-filepaths היא תמיד רשימה (גם אם היה רק נתיב אחד)
        if isinstance(filepaths, str):
            filepaths = [filepaths]
        # --- בדיקה אם קיבלנו רשימה בכלל ---
        if not isinstance(filepaths, list):
            print(f"  WARNING: Expected 'path' for label {label} to be a string or list, got {type(filepaths)}. Skipping label.")
            continue

        if not filepaths: # אם הרשימה ריקה, דלג על התווית
             print(f"  No file paths provided for label {label}. Skipping.")
             continue

        print(f"\nProcessing Label {label} ({CLASS_NAMES[label] if label < len(CLASS_NAMES) else 'Unknown'}) ({len(filepaths)} files)...")

        # <<< --- לולאה פנימית על כל נתיב ברשימה --- >>>
        for filepath in filepaths:
            # --- >>> התיקון: בדיקת סוג המשתנה filepath <<< ---
            if not isinstance(filepath, (str, bytes, os.PathLike)):
                print(f"\n  ERROR: Expected filepath inside list to be a string/path, but got {type(filepath)}: {filepath}. Skipping this entry.")
                continue # דלג על הערך הזה ברשימה והמשך לערך הבא
            # --- >>> סוף התיקון <<< ---

            # העבר את ההדפסה לכאן כדי שתודפס לפני הבדיקה אם הקובץ קיים
            print(f"  Processing file: {os.path.basename(filepath)} (Group ID: {group_counter})...", end='') # שימוש ב-group_counter הנוכחי

            # בדוק אם הקובץ קיים (עכשיו filepath מובטח להיות מסוג תקין)
            if not os.path.exists(filepath):
                print(f" ERROR: File not found! Skipping.")
                # אין צורך ב-group_counter += 1 כאן, זה יקרה בסוף עיבוד הקבוצה
                continue # דלג על הקובץ הספציפי הזה

            # המשך עיבוד הקובץ אם הוא קיים ותקין
            df = None; data = None; windows_created_count = 0
            try:
                # נסה לקרוא את ה-CSV
                try: df = pd.read_csv(filepath, header=None, skipinitialspace=True, **extra_args)
                except Exception: # אם נכשל, נסה לדלג על שורה
                     try: df = pd.read_csv(filepath, header=None, skiprows=1, skipinitialspace=True, **extra_args)
                     except Exception as e_read: print(f" ERROR reading CSV {filepath}: {e_read}. Skipping file."); continue # דלג על כל הקובץ

                # בדיקות תקינות DataFrame
                if df is None or df.empty: print(" Empty file/read failed. Skipping file."); continue
                df = df.dropna();
                if df.empty: print(" Empty after dropna. Skipping file."); continue;

                # המרה למספרים
                try:
                   data = df.values
                   if not np.issubdtype(data.dtype, np.number): data = data.astype(float)
                except ValueError as e_type: print(f" ERROR converting to numeric: {e_type}. Skipping file."); continue

                # בדיקות גודל
                if data.shape[0] < window_size or data.ndim < 2 or data.shape[1] == 0: print(f" Insufficient data shape {data.shape}. Skipping file."); continue

                # עיבודים (padding, filtering, windowing, spectrogram)
                data_padded = pad_channels(data)
                try: filtered_data = filtfilt(b, a, data_padded, axis=0)
                except Exception as e_filter: print(f" ERROR filtering data: {e_filter}. Skipping file."); continue

                windows, labels_w = create_windows(filtered_data, label)
                if not windows: print(" No windows created. Skipping file."); continue

                windows_created_count = len(windows); window_processed_count = 0
                for i, w in enumerate(windows):
                    try:
                        signal_to_process = w[:, 0].astype(np.float32)
                        mel_spec_norm = create_mel_spectrogram(signal_to_process)
                        if mel_spec_norm is None: continue
                        resized_spec = cv2.resize(mel_spec_norm, (target_img_size[1], target_img_size[0]), interpolation=cv2.INTER_LINEAR)
                        spec_3_channels = np.stack([resized_spec]*3, axis=0)
                        if not np.isfinite(spec_3_channels).all(): continue

                        # הוסף לרשימות הגלובליות
                        all_specs.append(spec_3_channels)
                        all_labels.append(labels_w[i])
                        all_groups.append(group_counter) # השתמש ב-group_counter הנוכחי
                        items_created_count += 1; window_processed_count += 1
                    except Exception as e_window: pass # אפשר להדפיס שגיאת חלון לדיבאג

                print(f" -> Processed {window_processed_count}/{windows_created_count} windows.")

            except Exception as e_file:
                print(f" UNEXPECTED ERROR processing file {filepath}: {e_file}")
                traceback.print_exc()
        # --- סוף הלולאה הפנימית ---

        # קדם את מונה הקבוצות *אחרי* שכל הקבצים עבור התווית הנוכחית עובדו
        # (בהנחה שכל קובץ הוא קבוצה, כפי שהיה בקוד המקורי)
        # אם קבצים שונים תחת אותה תווית צריכים קבוצות שונות, צריך לשנות את הלוגיקה הזו.
        # כרגע, אם label 0 הוא 'man.csv' ו-label 1 הוא 'car.csv',
        # כל הדגימות מ-man.csv יקבלו group=0, וכל הדגימות מ-car.csv יקבלו group=1.
        # אם ל-label 0 היו שני קבצים, שניהם היו מקבלים group=0 בלוגיקה הנוכחית.
        # אם אתה רוצה שכל קובץ יהיה קבוצה נפרדת, צריך להעביר את group_counter += 1
        # לסוף הלולאה הפנימית (for filepath in filepaths).
        # --> אני אשאיר את זה כפי שהיה במקור (קבוצה לכל תווית), אבל הוספתי הערה.
        group_counter += 1 # עבור התווית הבאה

    # --- סוף הלולאה החיצונית ---
    print(f"\nFinished loading all labels. Total items created: {items_created_count}")
    if not all_specs: print("WARNING: No spectrogram items were created successfully.")

    # המרה למערכי numpy
    specs_np = np.array(all_specs, dtype=np.float32) if all_specs else np.array([])
    labels_np = np.array(all_labels, dtype=np.int64) if all_labels else np.array([], dtype=np.int64)
    groups_np = np.array(all_groups, dtype=np.int64) if all_groups else np.array([], dtype=np.int64)

    del all_specs, all_labels, all_groups # שחרור זיכרון

    return specs_np, labels_np, None, groups_np # החזרת None עבור signals
# --- סוף פונקציית טעינת נתונים מתוקנת ---
