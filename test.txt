def pad_channels(data, target_channels=max_channels):
    # ... (כמו קודם) ...
    if data.ndim == 1: data = data.reshape(-1, 1)
    if data.shape[1] == target_channels: return data
    elif data.shape[1] < target_channels: padding = np.zeros((data.shape[0], target_channels - data.shape[1]), dtype=data.dtype); return np.hstack((data, padding))
    else: return data[:, :target_channels]

def create_windows(signal, label, window_size=window_size, stride=stride):
    # ... (כמו קודם) ...
    X_windows, y_labels = [], [];
    if signal.shape[0] < window_size: return [], []
    for start in range(0, signal.shape[0] - window_size + 1, stride): X_windows.append(signal[start : start + window_size, :]); y_labels.append(label)
    return X_windows, y_labels

def create_mel_spectrogram(window_1d, sr=int(fs), n_fft=N_FFT, hop_length=HOP_LENGTH, n_mels=N_MELS):
    # ... (כמו קודם, עם בדיקות) ...
    try:
        signal = window_1d.astype(np.float32); spec_width = int(np.floor(len(signal) / hop_length) + 1)
        if np.all(signal == signal[0]): return None # החזר None אם אות קבוע
        mel_spec = librosa.feature.melspectrogram(y=signal, sr=sr, n_fft=n_fft, hop_length=hop_length, n_mels=n_mels, fmin=1.0, fmax=min(highcut*1.2, sr/2 - 1))
        mel_spec_db = librosa.power_to_db(mel_spec, ref=np.max); min_db, max_db = mel_spec_db.min(), mel_spec_db.max()
        if max_db <= min_db: return None # החזר None אם אין טווח
        mel_spec_norm = (mel_spec_db - min_db) / (max_db - min_db); current_width = mel_spec_norm.shape[1]
        if current_width < spec_width: mel_spec_norm = np.pad(mel_spec_norm, ((0,0), (0, spec_width - current_width)), mode='constant')
        elif current_width > spec_width: mel_spec_norm = mel_spec_norm[:, :spec_width]
        if not np.isfinite(mel_spec_norm).all(): return None # החזר None אם יש NaN/inf
        return mel_spec_norm.astype(np.float32)
    except Exception as e:
        # print(f"Debug - Error in create_mel_spectrogram: {e}") # אפשר להוסיף להדפסת שגיאות
        return None # החזר None במקרה של שגיאה כללית

# <<< פונקציית טעינה משופרת שמחזירה גם Groups >>>
def load_data_specs_signals_and_groups(file_map, target_img_size=(IMG_HEIGHT_FINAL, IMG_WIDTH_FINAL)):
    all_specs, all_labels, all_signals, all_groups = [], [], [], [] # הוסף רשימה ל-groups
    try: b, a = butter(order, [lowcut/(fs/2), highcut/(fs/2)], btype='band')
    except ValueError as e: print(f"CRITICAL FILTER ERROR: {e}. Aborting."); return [], [], [], []
    print(f"Loading data, signals, and groups...")
    file_count = 0; items_created_count = 0; processed_files = 0
    group_counter = 0 # מונה לקבוצות ייחודיות

    for label, info in file_map.items():
        filepaths = info.get('path', []); extra_args = info.get('args', {})
        if isinstance(filepaths, str): filepaths = [filepaths]
        if not filepaths: continue
        file_count += len(filepaths)
        print(f"\nProcessing Label {label} ({len(filepaths)} files)...")

        for filepath in filepaths:
            processed_files += 1
            # <<< השתמש ב-group_id ייחודי לכל קובץ >>>
            group_id = group_counter # מזהה מספרי פשוט
            print(f"  Processing file: {filepath} (Group ID: {group_id})...")
            if not os.path.exists(filepath): print(f"    ERROR: File not found! Skipping."); continue

            df = None; data = None; windows_created_count = 0
            try:
                # ... (קוד טעינת CSV ועיבוד ראשוני כמו בקוד הקודם, עם הדפסות דיבאג) ...
                try: df = pd.read_csv(filepath, header=None, skipinitialspace=True, **extra_args)
                except Exception: df = pd.read_csv(filepath, header=None, skiprows=1, skipinitialspace=True, **extra_args)
                if df is None or df.empty: continue
                df = df.dropna();
                if df.empty: continue; data = df.values
                if not np.issubdtype(data.dtype, np.number): data = data.astype(float)
                if data.shape[0] < window_size or data.ndim < 2 or data.shape[1] == 0: continue
                data_padded = pad_channels(data)
                windows, labels_w = create_windows(data_padded, label)
                if not windows: continue
                windows_created_count = len(windows)

                window_processed_count = 0
                for i, w in enumerate(windows):
                    signal_to_process = None; spec_3_channels = None
                    try:
                        filtered_w = filtfilt(b, a, w, axis=0); signal_to_process = filtered_w[:, 0].astype(np.float32)
                        mel_spec_norm = create_mel_spectrogram(signal_to_process)
                        # <<< דלג על החלון אם הספקטרוגרמה לא נוצרה >>>
                        if mel_spec_norm is None: continue
                        resized_spec = cv2.resize(mel_spec_norm, (target_img_size[1], target_img_size[0]), interpolation=cv2.INTER_LINEAR)
                        spec_3_channels = np.stack([resized_spec]*3, axis=0)
                        if not np.isfinite(spec_3_channels).all(): continue

                        # --- הוסף גם את ה-group_id ---
                        all_signals.append(signal_to_process)
                        all_specs.append(spec_3_channels)
                        all_labels.append(labels_w[i])
                        all_groups.append(group_id) # הוסף את מזהה הקבוצה
                        items_created_count += 1
                        window_processed_count += 1
                    except Exception as e_window: print(f"        ERROR processing window {i} from file {filepath}: {e_window}"); pass
                print(f"    Finished processing {window_processed_count}/{windows_created_count} windows for this file.")

            except Exception as e_file: print(f"    UNEXPECTED ERROR processing file {filepath}: {e_file}")

            # <<< קדם את מונה הקבוצות רק אם הצלחנו לעבד את הקובץ (אפילו אם לא כל החלונות עברו) >>>
            # שינוי קל: נקדם תמיד כדי שלכל קובץ יהיה ID ייחודי, גם אם הוא נכשל לגמרי
            group_counter += 1


    print(f"\nFinished loading all files. Processed {processed_files} files. Total items created: {items_created_count}")
    if not all_specs: print("WARNING: No items were created successfully from any file.")
    specs_np = np.array(all_specs, dtype=np.float32) if all_specs else np.array([])
    labels_np = np.array(all_labels, dtype=np.int64) if all_labels else np.array([], dtype=np.int64)
    signals_np = np.array(all_signals, dtype=np.float32) if all_signals else np.array([])
    groups_np = np.array(all_groups) if all_groups else np.array([]) # המר גם את הקבוצות
    return specs_np, labels_np, signals_np, groups_np # <<< החזר 4 ערכים >>>

class SpecAugment(nn.Module): # כמו קודם
    def __init__(self, freq_mask_param=20, time_mask_param=40, num_freq_masks=1, num_time_masks=1):
        super().__init__(); self.freq_mask_param = freq_mask_param; self.time_mask_param = time_mask_param
        self.num_freq_masks = num_freq_masks; self.num_time_masks = num_time_masks
    def forward(self, spec):
        if not isinstance(spec, torch.Tensor): raise TypeError("Input must be a Tensor."); spec_aug = spec.clone(); _, F, T = spec_aug.shape; fm_param = min(self.freq_mask_param, F // 2); tm_param = min(self.time_mask_param, T // 2)
        for _ in range(self.num_freq_masks):
            if fm_param > 0: f = random.randrange(0, fm_param);
            if F-f > 0: f0 = random.randrange(0, F - f); spec_aug[:, f0:f0+f, :] = spec.mean()
        for _ in range(self.num_time_masks):
             if tm_param > 0: t = random.randrange(0, tm_param);
             if T-t > 0: t0 = random.randrange(0, T - t); spec_aug[:, :, t0:t0+t] = spec.mean()
        return spec_aug

class SpectrogramDataset(Dataset): # כמו קודם
    def __init__(self, specs_np, labels_np, transform=None):
        self.specs = torch.from_numpy(specs_np.astype(np.float32))
        self.labels = torch.from_numpy(labels_np.astype(np.int64))
        self.transform = transform
    def __len__(self): return len(self.labels)
    def __getitem__(self, idx):
        spec = self.specs[idx]; label = self.labels[idx]
        if self.transform: spec = self.transform(spec)
        return spec, label

def setup_efficientnet_model(model_name='efficientnet_b0', pretrained=True, num_classes=NUM_CLASSES, dropout_p=0.2): # כמו קודם
    weights = EfficientNet_B0_Weights.DEFAULT if pretrained else None; model_arch = efficientnet_b0(weights=weights)
    num_ftrs = model_arch.classifier[1].in_features; model_arch.classifier = nn.Sequential(nn.Dropout(p=dropout_p, inplace=True), nn.Linear(num_ftrs, num_classes))
    if pretrained:
        for param in model_arch.features.parameters(): param.requires_grad = False
        for param in model_arch.classifier.parameters(): param.requires_grad = True
    model_arch = model_arch.to(DEVICE); return model_arch
def train_epoch_effnet(model, dataloader, criterion, optimizer, device): # כמו קודם
    model.train(); running_loss = 0.0; correct_preds = 0; total_samples = 0;
    for inputs, labels in dataloader:
        try: inputs, labels = inputs.to(device), labels.to(device); optimizer.zero_grad(); outputs = model(inputs); loss = criterion(outputs, labels)
        if loss is None or torch.isnan(loss): continue; preds = outputs.max(1)[1]; loss.backward(); optimizer.step(); running_loss += loss.item() * inputs.size(0); correct_preds += (preds == labels).sum().item(); total_samples += labels.size(0)
        except Exception as e: print(f"  Error train batch: {e}"); continue
    if total_samples == 0: return 0.0, 0.0; return running_loss / total_samples, correct_preds / total_samples
def evaluate_model_effnet(model, dataloader, criterion, device): # כמו קודם
    model.eval(); running_loss = 0.0; correct_preds = 0; total_samples = 0; all_preds = []; all_labels = []
    with torch.no_grad():
        for inputs, labels in dataloader:
            try: inputs, labels = inputs.to(device), labels.to(device); outputs = model(inputs); loss = criterion(outputs, labels);
            if loss is None or torch.isnan(loss): continue; preds = outputs.max(1)[1]; running_loss += loss.item() * inputs.size(0); correct_preds += (preds == labels).sum().item(); total_samples += labels.size(0); all_preds.extend(preds.cpu().numpy()); all_labels.extend(labels.cpu().numpy())
            except Exception as e: print(f"  Error eval batch: {e}"); continue
    if total_samples == 0: return 0.0, 0.0, [], []; return running_loss / total_samples, correct_preds / total_samples, all_labels, all_preds

print("\n--- Defining Optuna Objective Function (with StratifiedGroupKFold) ---")

def objective(trial):
    start_trial_time = time.time()
    print(f"\n{'='*15} Optuna Trial {trial.number} Starting {'='*15}")

    # --- דגימת היפרפרמטרים ---
    lr = trial.suggest_float("lr", 1e-4, 5e-3, log=True) # הרחב מעט טווח LR
    dropout_p = trial.suggest_float("dropout_p", 0.2, 0.5, step=0.1) # צמצם טווח דרופאאוט
    weight_decay = trial.suggest_float("weight_decay", 1e-5, 1e-3, log=True)
    specaug_freq = trial.suggest_int("specaug_freq", 5, 25) # הקטן מעט טווחים
    specaug_time = trial.suggest_int("specaug_time", 10, 50)
    specaug_f_masks = trial.suggest_int("specaug_f_masks", 1, 2)
    specaug_t_masks = trial.suggest_int("specaug_t_masks", 1, 2)

    print(f"  Trial Params: LR={lr:.1e}, Dropout={dropout_p:.2f}, WD={weight_decay:.1e}, "
          f"SpecAugF={specaug_freq}, SpecAugT={specaug_time}, N_F={specaug_f_masks}, N_T={specaug_t_masks}")

    # --- הגדרת טרנספורמציות אימון ספציפיות ל-trial ---
    current_train_transforms = transforms.Compose([
        SpecAugment(freq_mask_param=specaug_freq, time_mask_param=specaug_time,
                    num_freq_masks=specaug_f_masks, num_time_masks=specaug_t_masks),
        transforms.Normalize(mean=imagenet_mean, std=imagenet_std)
    ])

    # --- Cross-Validation עם StratifiedGroupKFold ---
    # <<< שימוש ב-StratifiedGroupKFold למניעת דליפה >>>
    kfold = StratifiedGroupKFold(n_splits=N_SPLITS_OPT, shuffle=True, random_state=trial.number + SEED_OFFSET)
    fold_val_accs = []         # לאחסון Val Acc מכל fold (עכשיו זה מדד אמין יותר)
    fold_new_data_accs = []    # לאחסון Test Acc (נתונים חדשים) מכל fold שעמד בתנאים
    # <<< לא נשמור כאן מודלים מלאים, רק את המידע עליהם >>>
    qualified_fold_info_for_trial = [] # רשימה של מילונים עם תוצאות הפולדים המוצלחים

    # <<< השתמש בנתונים הגלובליים שנטענו קודם >>>
    X = orig_specs_np
    y = orig_labels_np
    groups = orig_groups_np

    for fold, (train_ids, val_ids) in enumerate(kfold.split(X, y, groups)):
        fold_start_time = time.time()
        print(f"  --- Fold {fold+1}/{N_SPLITS_OPT} ---")
        # הדפס סטטיסטיקה על הקבוצות בכל פיצול - לבדיקה
        train_groups, val_groups = np.unique(groups[train_ids]), np.unique(groups[val_ids])
        print(f"    Train groups: {len(train_groups)}, Val groups: {len(val_groups)}, Overlap: {len(set(train_groups) & set(val_groups))}") # צריך להיות 0

        # --- הכנת נתונים ל-fold ---
        X_train_fold, y_train_fold = X[train_ids], y[train_ids]
        X_val_fold, y_val_fold = X[val_ids], y[val_ids]
        train_dataset = SpectrogramDataset(X_train_fold, y_train_fold, transform=current_train_transforms)
        val_dataset = SpectrogramDataset(X_val_fold, y_val_fold, transform=val_test_transforms)
        train_loader = DataLoader(train_dataset, batch_size=BATCH_SIZE_OPT, shuffle=True, num_workers=0, pin_memory=True)
        val_loader = DataLoader(val_dataset, batch_size=BATCH_SIZE_OPT, shuffle=False, num_workers=0, pin_memory=True)

        # --- הגדרת מודל, קריטריון ואופטימייזר ---
        model = setup_efficientnet_model(model_name=MODEL_NAME_OPT, pretrained=True, num_classes=NUM_CLASSES, dropout_p=dropout_p)
        criterion = nn.CrossEntropyLoss()
        # אופטימיזציה רק של פרמטרי ה-classifier (אם pretrained=True)
        params_to_optimize = model.classifier.parameters() if any(p.requires_grad for p in model.features.parameters()) == False else model.parameters()
        optimizer = optim.AdamW(params_to_optimize, lr=lr, weight_decay=weight_decay)

        # --- אימון ה-fold עם Early Stopping ---
        best_val_acc_fold = 0.0; current_best_fold_model_state = None; epochs_no_improve = 0; best_epoch = -1
        history_current_fold = {'train_loss': [], 'train_acc': [], 'val_loss': [], 'val_acc': []}
        best_fold_val_labels = []; best_fold_val_preds = []

        for epoch in range(1, EPOCHS_OPT + 1):
            train_loss, train_acc = train_epoch_effnet(model, train_loader, criterion, optimizer, device)
            val_loss, val_acc, fold_val_labels_epoch, fold_val_preds_epoch = evaluate_model_effnet(model, val_loader, criterion, device)
            print(f"    Epoch {epoch}/{EPOCHS_OPT} | Train Loss: {train_loss:.4f}, Acc: {train_acc:.4f} | Val Loss: {val_loss:.4f}, Acc: {val_acc:.4f} | Time: {time.time() - time.time():.1f}s")
            history_current_fold['train_loss'].append(train_loss); history_current_fold['train_acc'].append(train_acc); history_current_fold['val_loss'].append(val_loss); history_current_fold['val_acc'].append(val_acc)
            if val_acc > best_val_acc_fold:
                best_val_acc_fold = val_acc; current_best_fold_model_state = deepcopy(model.state_dict()); epochs_no_improve = 0; best_epoch = epoch
                best_fold_val_labels = fold_val_labels_epoch; best_fold_val_preds = fold_val_preds_epoch
            else: epochs_no_improve += 1
            if epochs_no_improve >= PATIENCE_OPT: print(f"    Early stopping at epoch {epoch}. Best val_acc: {best_val_acc_fold:.4f}"); break

        # --- הערכה על נתונים חדשים (אם רלוונטי) ---
        current_fold_new_data_acc = 0.0
        fold_new_data_labels = None; fold_new_data_preds = None

        if current_best_fold_model_state and new_specs_np.size > 0:
            print(f"    Fold {fold+1} completed. Best Val Acc (Leak-Free): {best_val_acc_fold:.4f} at epoch {best_epoch}.")
            if MIN_VAL_ACC_THRESHOLD <= best_val_acc_fold <= MAX_VAL_ACC_THRESHOLD:
                print(f"      Val Acc within threshold. Evaluating on new data...")
                model.load_state_dict(current_best_fold_model_state)
                new_dataset = SpectrogramDataset(new_specs_np, new_labels_np, transform=val_test_transforms)
                new_loader = DataLoader(new_dataset, batch_size=BATCH_SIZE_OPT, shuffle=False, num_workers=0, pin_memory=True)
                try:
                    new_loss, new_acc, fold_new_data_labels, fold_new_data_preds = evaluate_model_effnet(model, new_loader, criterion, device)
                    current_fold_new_data_acc = new_acc # שמור את הדיוק על הנתונים החדשים
                    print(f"        -> New Data Eval: Loss={new_loss:.4f}, Accuracy={current_fold_new_data_acc:.4f}")
                    # <<< שמור מידע על ה-fold המוצלח (ללא state_dict) >>>
                    qualified_fold_info_for_trial.append({
                        'fold': fold + 1,
                        'val_acc': best_val_acc_fold,
                        'new_acc': current_fold_new_data_acc,
                        'epoch': best_epoch,
                        # 'state_dict': current_best_fold_model_state, # <<< לא שומרים כאן! >>>
                        'history': history_current_fold,
                        'val_labels': best_fold_val_labels, 'val_preds': best_fold_val_preds,
                        'new_labels': fold_new_data_labels, 'new_preds': fold_new_data_preds,
                    })
                except Exception as e_eval_new: print(f"        ERROR during new data evaluation: {e_eval_new}")
            else: print(f"      Val Acc outside threshold. Skipping new data evaluation.")
        elif new_specs_np.size == 0: print(f"    No new data loaded. Skipping new data evaluation.")
        else: print(f"    No best model state found for fold {fold+1}. Skipping new data evaluation.")

        fold_val_accs.append(best_val_acc_fold) # שמור תמיד את ה-Val Acc (האמין)
        fold_new_data_accs.append(current_fold_new_data_acc) # שמור 0 אם לא הוערך

        # --- דיווח ל-Optuna ובדיקת Pruning ---
        # דווח על המדד המשני (Val Acc) כדי לאפשר Pruning מוקדם
        trial.report(best_val_acc_fold, fold)
        if trial.should_prune(): print("  Trial pruned based on validation accuracy."); raise optuna.exceptions.TrialPruned()
        print(f"  --- Fold {fold+1} Finished. Time: {time.time() - fold_start_time:.1f}s ---")

    # --- חישוב תוצאה סופית ל-Trial ---
    average_val_acc = np.mean(fold_val_accs) if fold_val_accs else 0.0
    # <<< המדד שרוצים למקסם: ממוצע הדיוק על הנתונים החדשים (מהפולדים שעמדו בתנאי) >>>
    qualified_new_accs = [f['new_acc'] for f in qualified_fold_info_for_trial]
    average_acc_new_data = np.mean(qualified_new_accs) if qualified_new_accs else 0.0

    print(f"\n--- Finished Trial {trial.number} ---")
    print(f"  Avg Val Acc (Leak-Free): {average_val_acc:.4f}")
    print(f"  Avg Test Acc (Normalized Data, qualified folds only): {average_acc_new_data:.4f}")
    print(f"  Total Trial Time: {time.time() - start_trial_time:.1f}s")
    print(f"{'='*40}")

    # --- שמירת מידע על ה-Trial ב-User Attributes ---
    if average_acc_new_data > 0 and qualified_fold_info_for_trial:
        trial_summary = {
            'trial_num': trial.number, 'params': trial.params,
            'avg_val_acc': average_val_acc, 'avg_test_acc_new': average_acc_new_data,
            'qualified_folds_metrics': [{'fold': f['fold'], 'val_acc': f['val_acc'], 'new_acc': f['new_acc']} for f in qualified_fold_info_for_trial]
        }
        # <<< שמור את התוצאות המלאות של הפולדים (כולל הכל חוץ מ-state_dict) >>>
        trial.set_user_attr("trial_summary", trial_summary)
        trial.set_user_attr("qualified_fold_results", qualified_fold_info_for_trial)

        # --- חשוב: שמור את ה-state_dict של המודל הטוב ביותר *מה-trial הזה* בנפרד ---
        # בחר את ה-fold הכי טוב *בתוך ה-trial הנוכחי* לפי הקריטריון שקבענו
        qualified_fold_info_for_trial.sort(key=lambda x: (-x['val_acc'], -x['new_acc'])) # קודם Val Acc גבוה
        best_fold_in_trial = qualified_fold_info_for_trial[0]
        # טען מחדש את המצב הטוב ביותר מה-fold הזה
        model.load_state_dict(best_fold_in_trial['state_dict'])
        # שמור את ה-state_dict ב-user_attr נוסף
        trial.set_user_attr("best_fold_state_dict", deepcopy(best_fold_in_trial['state_dict'])) # שמור עותק עמוק
        trial.set_user_attr("best_fold_num_in_trial", best_fold_in_trial['fold'])


    # <<< החזר את המדד הראשי למקסום >>>
    return average_acc_new_data
