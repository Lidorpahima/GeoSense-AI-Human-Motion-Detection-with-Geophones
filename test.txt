def pad_channels(data, target_channels):
    if data.ndim == 1:
        data = data.reshape(-1, 1)
    if data.shape[1] < target_channels:
        padding = np.zeros((data.shape[0], target_channels - data.shape[1]), dtype=data.dtype)
        return np.hstack((data, padding))
    else:
        return data[:, :target_channels]

def create_windows(signal, label, window_size=window_size, stride=stride):
    X_windows, y_labels = [], []
    if signal.shape[0] < window_size:
        return [], []
    for start in range(0, signal.shape[0] - window_size + 1, stride):
        X_windows.append(signal[start: start + window_size, :])
        y_labels.append(label)
    return X_windows, y_labels

def create_stft_spectrogram(signal_1d, sr=fs, n_fft=N_FFT, hop_length=HOP_LENGTH):
    try:
        signal_float = signal_1d.astype(np.float32)
        f, t, Sxx = spectrogram(signal_float, fs=sr, nperseg=n_fft, noverlap=n_fft - hop_length, nfft=n_fft)
        # הגבלת תדר עד highcut
        freq_idx = np.where(f <= highcut)[0]
        if len(freq_idx) == 0:
            return None
        Sxx = Sxx[freq_idx, :]
        Sxx_db = 10 * np.log10(Sxx + 1e-9)
        Sxx_norm = (Sxx_db - Sxx_db.min()) / (Sxx_db.max() - Sxx_db.min() + 1e-9)
        return Sxx_norm.astype(np.float32)
    except Exception as e:
        print(f"Error in STFT: {e}")
        return None

def extract_statistical_features(signal_1d, sr=fs):
    try:
        features = []
        features.append(np.mean(signal_1d))
        features.append(np.std(signal_1d))
        features.append(np.sqrt(np.mean(signal_1d**2)))
        features.append(librosa.feature.zero_crossing_rate(y=signal_1d).mean())
        stft_result = librosa.stft(signal_1d.astype(np.float32), n_fft=N_FFT, hop_length=HOP_LENGTH)
        spec_mag = np.abs(stft_result)
        features.append(librosa.feature.spectral_centroid(S=spec_mag, sr=sr).mean())
        features.append(librosa.feature.spectral_bandwidth(S=spec_mag, sr=sr).mean())
        features.append(librosa.feature.spectral_rolloff(S=spec_mag, sr=sr, roll_percent=0.85).mean())
        features.append(librosa.feature.spectral_flatness(S=spec_mag).mean())
        mfccs = librosa.feature.mfcc(y=signal_1d.astype(np.float32), sr=sr, n_mfcc=13)
        features.extend(np.mean(mfccs, axis=1))
        return np.array(features, dtype=np.float32)
    except Exception as e:
        print(f"Error extracting statistical features: {e}")
        num_features = 8 + 13
        return np.zeros(num_features, dtype=np.float32)

NUM_STAT_FEATURES = 8 + 13

def load_data_hybrid_by_file(file_map, target_img_size=(IMG_HEIGHT, IMG_WIDTH)):
    all_specs, all_stat_features, all_labels, all_groups = [], [], [], []
    try:
        b, a = butter(order, [lowcut/(fs/2), highcut/(fs/2)], btype='band')
    except Exception as e:
        print(f"Filter error: {e}")
        return np.array([]), np.array([]), np.array([]), np.array([])
    
    group_counter = 0
    print("Loading data by file and extracting hybrid features...")
    for label, info in file_map.items():
        filepaths = info['path']
        if isinstance(filepaths, str):
            filepaths = [filepaths]
        for filepath in filepaths:
            try:
                df = pd.read_csv(filepath, header=None)
                df = df.dropna()
                data = df.values
                data = pad_channels(data, max_channels)
                if data.shape[0] < window_size:
                    continue
                windows, labels = create_windows(data, label)
                for w in windows:
                    try:
                        filtered_w = filtfilt(b, a, w, axis=0)
                        # בוחרים את הערוץ הראשון
                        signal_to_process = filtered_w[:, 0]
                        spec = create_stft_spectrogram(signal_to_process)
                        if spec is None:
                            continue
                        resized_spec = cv2.resize(spec, (target_img_size[1], target_img_size[0]), interpolation=cv2.INTER_LINEAR)
                        spec_3ch = np.stack([resized_spec]*3, axis=0)  # צורת (3, H, W)
                        stat_feats = extract_statistical_features(signal_to_process)
                        all_specs.append(spec_3ch)
                        all_stat_features.append(stat_feats)
                        all_labels.append(label)
                        all_groups.append(group_counter)
                    except Exception as inner_e:
                        print(f"Error processing window in file {filepath}: {inner_e}")
                group_counter += 1
            except Exception as e:
                print(f"Error loading file {filepath}: {e}")
    print(f"Total samples loaded: {len(all_specs)}")
    return np.array(all_specs, dtype=np.float32), np.array(all_stat_features, dtype=np.float32), np.array(all_labels, dtype=np.int64), np.array(all_groups, dtype=np.int64)

class SpecAugment(nn.Module):
    def __init__(self, freq_mask_param=10, time_mask_param=20, num_freq_masks=1, num_time_masks=1):
        super(SpecAugment, self).__init__()
        self.freq_mask_param = freq_mask_param
        self.time_mask_param = time_mask_param
        self.num_freq_masks = num_freq_masks
        self.num_time_masks = num_time_masks
    
    def forward(self, spec):
        if not isinstance(spec, torch.Tensor):
            raise TypeError("Input must be a tensor")
        spec_aug = spec.clone()
        _, F, T = spec_aug.shape
        for _ in range(self.num_freq_masks):
            f = random.randrange(0, self.freq_mask_param)
            f0 = random.randrange(0, max(1, F - f))
            spec_aug[:, f0:f0+f, :] = spec_aug.mean()
        for _ in range(self.num_time_masks):
            t = random.randrange(0, self.time_mask_param)
            t0 = random.randrange(0, max(1, T - t))
            spec_aug[:, :, t0:t0+t] = spec_aug.mean()
        return spec_aug

class HybridDataset(Dataset):
    def __init__(self, specs, stat_feats, labels, transform=None):
        self.specs = specs
        self.stat_feats = stat_feats
        self.labels = labels
        self.transform = transform
    
    def __len__(self):
        return len(self.labels)
    
    def __getitem__(self, idx):
        spec = self.specs[idx]      # צורה: (3, H, W)
        stat_feat = self.stat_feats[idx]
        label = self.labels[idx]
        spec_tensor = torch.tensor(spec, dtype=torch.float32)
        if self.transform:
            spec_tensor = self.transform(spec_tensor)
        stat_tensor = torch.tensor(stat_feat, dtype=torch.float32)
        return spec_tensor, stat_tensor, label

# --- 5. הגדרת המודל ההיברידי ---
import torch.nn.functional as F

class HybridGeoModel(nn.Module):
    def __init__(self, dropout_p=0.4, stat_feature_dim=64, combined_dim=128):
        super(HybridGeoModel, self).__init__()
        weights = MobileNet_V3_Small_Weights.DEFAULT
        self.cnn_backbone = mobilenet_v3_small(weights=weights)
        # הסרת שכבת הסיווג האחרונה
        cnn_out_features = self.cnn_backbone.classifier[3].in_features
        self.cnn_backbone.classifier = nn.Identity()
        # הקפאת חלק מהמשקולות
        for param in self.cnn_backbone.features.parameters():
            param.requires_grad = False
        # MLP עבור תכונות סטטיסטיות
        self.stat_mlp = nn.Sequential(
            nn.LayerNorm(NUM_STAT_FEATURES),
            nn.Linear(NUM_STAT_FEATURES, stat_feature_dim * 2),
            nn.ReLU(),
            nn.Dropout(dropout_p),
            nn.Linear(stat_feature_dim * 2, stat_feature_dim),
            nn.ReLU()
        )
        # שילוב תכונות
        self.dropout = nn.Dropout(dropout_p)
        self.fc_combined = nn.Linear(cnn_out_features + stat_feature_dim, combined_dim)
        self.fc_output   = nn.Linear(combined_dim, NUM_CLASSES)
    
    def forward(self, spec_input, stat_input):
        cnn_features = self.cnn_backbone(spec_input)
        stat_features = self.stat_mlp(stat_input)
        combined = torch.cat((cnn_features, stat_features), dim=1)
        x = self.dropout(combined)
        x = F.relu(self.fc_combined(x))
        x = self.dropout(x)
        output = self.fc_output(x)
        return output

# --- 6. פונקציות אימון והערכה ---
def train_epoch_hybrid(model, dataloader, criterion, optimizer, device):
    model.train()
    running_loss = 0.0
    correct = 0
    total = 0
    for specs, stat_feats, labels in dataloader:
        specs = specs.to(device)
        stat_feats = stat_feats.to(device)
        labels = labels.to(device)
        optimizer.zero_grad()
        outputs = model(specs, stat_feats)
        loss = criterion(outputs, labels)
        loss.backward()
        optimizer.step()
        running_loss += loss.item() * labels.size(0)
        preds = outputs.argmax(dim=1)
        correct += (preds == labels).sum().item()
        total += labels.size(0)
    epoch_loss = running_loss / total
    epoch_acc = correct / total
    return epoch_loss, epoch_acc

def evaluate_model_hybrid(model, dataloader, criterion, device):
    model.eval()
    running_loss = 0.0
    correct = 0
    total = 0
    all_labels = []
    all_preds = []
    with torch.no_grad():
        for specs, stat_feats, labels in dataloader:
            specs = specs.to(device)
            stat_feats = stat_feats.to(device)
            labels = labels.to(device)
            outputs = model(specs, stat_feats)
            loss = criterion(outputs, labels)
            running_loss += loss.item() * labels.size(0)
            preds = outputs.argmax(dim=1)
            correct += (preds == labels).sum().item()
            total += labels.size(0)
            all_labels.extend(labels.cpu().numpy())
            all_preds.extend(preds.cpu().numpy())
    epoch_loss = running_loss / total
    epoch_acc = correct / total
    return epoch_loss, epoch_acc, all_labels, all_preds

def plot_fold_results(title, history, true_labels, pred_labels):

# --- 7. פונקציית מטרה לאופטונה (עם GroupKFold) ---
# נשתמש ב-Optuna לחיפוש היפר-פרמטרים
STUDY_NAME = "hybrid_model_optuna"
STUDY_FILENAME = f"{STUDY_NAME}.pkl"
try:
    study = joblib.load(STUDY_FILENAME)
    print(f"Loaded study {STUDY_NAME}")
except FileNotFoundError:
    study = optuna.create_study(study_name=STUDY_NAME, direction="maximize")
    
def save_study_callback(study, trial):
    joblib.dump(study, STUDY_FILENAME)

def objective(trial):
    start_time = time.time()
    print(f"\n--- Optuna Trial {trial.number} ---")
    # דגימת היפר-פרמטרים
    lr         = trial.suggest_float("lr", 1e-4, 1e-2, log=True)
    dropout_p  = trial.suggest_float("dropout_p", 0.2, 0.6, step=0.1)
    weight_decay = trial.suggest_float("weight_decay", 1e-5, 1e-3, log=True)
    stat_dim   = trial.suggest_categorical("stat_dim", [32, 64, 128])
    comb_dim   = trial.suggest_categorical("comb_dim", [64, 128, 256])
    specaug_freq = trial.suggest_int("specaug_freq", 10, 30)
    specaug_time = trial.suggest_int("specaug_time", 20, 60)
    specaug_f_masks = trial.suggest_int("specaug_f_masks", 1, 2)
    specaug_t_masks = trial.suggest_int("specaug_t_masks", 1, 2)
    
    print(f"Params: lr={lr:.1e}, dropout={dropout_p}, wd={weight_decay:.1e}, stat_dim={stat_dim}, comb_dim={comb_dim}")
    
    # עדכון טרנספורמציות האימון לפי הפרמטרים
    current_train_transforms = transforms.Compose([
        SpecAugment(freq_mask_param=specaug_freq, time_mask_param=specaug_time,
                    num_freq_masks=specaug_f_masks, num_time_masks=specaug_t_masks),
        transforms.Normalize(mean=imagenet_mean, std=imagenet_std)
    ])
    
    # שימוש ב-GroupKFold (מניעת Data Leakage לפי קובץ)
    gkf = GroupKFold(n_splits=N_SPLITS)
    fold_val_accs = []
    fold_recalls  = []
    fold_best_models = []
    
    for fold, (train_idx, val_idx) in enumerate(gkf.split(orig_specs, orig_labels, groups=orig_groups)):
        print(f"  Fold {fold+1}/{N_SPLITS}")
        # חלוקת נתונים לפי אינדקסים
        X_spec_train = orig_specs[train_idx]
        X_stat_train = orig_stat_feats[train_idx]
        y_train      = orig_labels[train_idx]
        X_spec_val   = orig_specs[val_idx]
        X_stat_val   = orig_stat_feats[val_idx]
        y_val        = orig_labels[val_idx]
        
        # נרמול התכונות הסטטיסטיות
        scaler = StandardScaler()
        X_stat_train = scaler.fit_transform(X_stat_train)
        X_stat_val   = scaler.transform(X_stat_val)
        
        # יצירת דאטהסטים
        train_dataset = HybridDataset(X_spec_train, X_stat_train, y_train, transform=current_train_transforms)
        val_dataset   = HybridDataset(X_spec_val, X_stat_val, y_val, transform=val_transforms)
        train_loader = DataLoader(train_dataset, batch_size=BATCH_SIZE, shuffle=True, num_workers=0)
        val_loader   = DataLoader(val_dataset, batch_size=BATCH_SIZE, shuffle=False, num_workers=0)
        
        # הגדרת המודל
        model = HybridGeoModel(dropout_p=dropout_p, stat_feature_dim=stat_dim, combined_dim=comb_dim).to(device)
        criterion = nn.CrossEntropyLoss()
        optimizer = optim.AdamW(filter(lambda p: p.requires_grad, model.parameters()), lr=lr, weight_decay=weight_decay)
        
        best_val_acc = 0.0
        best_val_loss = float('inf')
        best_model_state = None
        epochs_no_improve = 0
        history = {'train_loss': [], 'train_acc': [], 'val_loss': [], 'val_acc': []}
        for epoch in range(1, EPOCHS+1):
            train_loss, train_acc = train_epoch_hybrid(model, train_loader, criterion, optimizer, device)
            val_loss, val_acc, val_labels, val_preds = evaluate_model_hybrid(model, val_loader, criterion, device)
            history['train_loss'].append(train_loss)
            history['train_acc'].append(train_acc)
            history['val_loss'].append(val_loss)
            history['val_acc'].append(val_acc)
            print(f"    Epoch {epoch}: Train Acc={train_acc:.4f}, Val Acc={val_acc:.4f}")
            if val_loss < best_val_loss:
                best_val_loss = val_loss
                best_val_acc = val_acc
                best_model_state = deepcopy(model.state_dict())
                epochs_no_improve = 0
            else:
                epochs_no_improve += 1
            if epochs_no_improve >= PATIENCE:
                print("    Early stopping")
                break
        fold_val_accs.append(best_val_acc)
        
        # הערכת המודל על דאטה חדש אם קיים
        new_data_recall = 0.0
        if new_specs.size > 0:
            new_stat_scaled = scaler.transform(new_stat_feats)
            new_dataset = HybridDataset(new_specs, new_stat_scaled, new_labels, transform=val_transforms)
            new_loader = DataLoader(new_dataset, batch_size=BATCH_SIZE, shuffle=False, num_workers=0)
            _, _, new_true, new_pred = evaluate_model_hybrid(model, new_loader, criterion, device)
            new_data_recall = recall_score(new_true, new_pred, average='macro', zero_division=0)
            print(f"    New data recall: {new_data_recall:.4f}")
        fold_recalls.append(new_data_recall)
        fold_best_models.append({
            'fold': fold+1,
            'val_acc': best_val_acc,
            'new_recall': new_data_recall,
            'state_dict': best_model_state,
            'history': history,
            'scaler': scaler,
            'val_labels': val_labels,
            'val_preds': val_preds
        })
    avg_val_acc = np.mean(fold_val_accs)
    avg_recall  = np.mean(fold_recalls)
    combined_score = 0.3 * avg_val_acc + 0.7 * avg_recall
    print(f"Trial {trial.number}: Avg Val Acc={avg_val_acc:.4f}, Avg New Recall={avg_recall:.4f}, Combined Score={combined_score:.4f}")
    
    trial.report(combined_score, step=0)
    if trial.should_prune():
        print("Trial pruned")
        raise optuna.exceptions.TrialPruned()
    
    trial.set_user_attr("fold_models", fold_best_models)
    return combined_score

def pad_channels(data, target_channels): 
    if data.ndim == 1: data = data.reshape(-1, 1)
    if data.shape[1] == target_channels: return data
    elif data.shape[1] < target_channels:
        padding = np.zeros((data.shape[0], target_channels - data.shape[1]), dtype=data.dtype)
        return np.hstack((data, padding))
    else: return data[:, :target_channels]

def create_windows(signal, label, window_size=window_size, stride=stride): 
    X_windows, y_labels = [], []
    if signal.shape[0] < window_size: return [], []
    for start in range(0, signal.shape[0] - window_size + 1, stride):
        X_windows.append(signal[start : start + window_size, :])
        y_labels.append(label)
    return X_windows, y_labels

def add_gaussian_noise_signal(signal_1d, noise_level=0.01):
    noise = np.random.normal(0, noise_level, signal_1d.shape)
    return (signal_1d + noise).astype(np.float32)

def time_stretch_signal(signal_1d, stretch_rate=1.0, target_len=window_size):
    try: 
        stretched = librosa.effects.time_stretch(signal_1d.astype(np.float32), rate=stretch_rate)
        current_len = len(stretched)
        if current_len == target_len: return stretched
        elif current_len > target_len: start = random.randint(0, current_len - target_len); return stretched[start : start + target_len]
        else: pad_len = target_len - current_len; pad_start = random.randint(0, pad_len); padded = np.zeros(target_len, dtype=np.float32); padded[pad_start : pad_start + current_len] = stretched; return padded
    except Exception: return signal_1d.astype(np.float32) 

def create_stft_spectrogram(window_1d, sr=int(fs), n_fft=N_FFT_STFT, hop_length=HOP_LENGTH_STFT): 
    try:
        signal = window_1d.astype(np.float32)
        f, t, Sxx = spectrogram(signal, fs=sr, nperseg=n_fft, noverlap=n_fft - hop_length, nfft=n_fft, detrend=False, scaling='spectrum') 
        freq_slice = np.where(f <= highcut * 1.1)[0]; If len(freq_slice) == 0: return np.zeros((10, 10), dtype=np.float32) 
        f = f[freq_slice]; Sxx = Sxx[freq_slice, :]
        if Sxx.size == 0: return np.zeros((len(f), 10), dtype=np.float32) 
        Sxx_db = 10 * np.log10(Sxx + 1e-9); min_db, max_db = np.percentile(Sxx_db, 5), np.percentile(Sxx_db, 95)
        if max_db <= min_db: max_db = min_db + 1e-6 
        spec_norm = np.clip((Sxx_db - min_db) / (max_db - min_db), 0, 1) 
        return spec_norm.astype(np.float32) 
    except Exception: return np.zeros((N_FREQ_BINS // 2, 10), dtype=np.float32) 

def extract_statistical_features(window_1d, sr=int(fs)):
    features = []; num_expected_features = 8 + 13 
    try:
        features.append(np.mean(window_1d)); features.append(np.std(window_1d)); features.append(np.sqrt(np.mean(window_1d**2))) 
        features.append(librosa.feature.zero_crossing_rate(y=window_1d).mean())
        stft_result = librosa.stft(window_1d.astype(np.float32), n_fft=N_FFT_STFT, hop_length=HOP_LENGTH_STFT)
        spec_mag = np.abs(stft_result)
        features.append(librosa.feature.spectral_centroid(S=spec_mag, sr=sr).mean())
        features.append(librosa.feature.spectral_bandwidth(S=spec_mag, sr=sr).mean())
        features.append(librosa.feature.spectral_rolloff(S=spec_mag, sr=sr, roll_percent=0.85).mean())
        features.append(librosa.feature.spectral_flatness(S=spec_mag).mean())
        mfccs = librosa.feature.mfcc(y=window_1d.astype(np.float32), sr=sr, n_mfcc=13)
        features.extend(np.mean(mfccs, axis=1)) 
        if len(features) != num_expected_features: raise ValueError("Incorrect feature count")
    except Exception as e: return np.zeros(num_expected_features, dtype=np.float32)
    return np.array(features, dtype=np.float32)

        if not isinstance(spec, torch.Tensor): raise TypeError("Input must be a Tensor.")
        spec_aug = spec.clone(); _, F, T = spec_aug.shape; fm_param = min(self.freq_mask_param, F // 2); tm_param = min(self.time_mask_param, T // 2)

        for _ in range(self.num_time_masks):
             if tm_param > 0: t = random.randrange(0, tm_param); t0 = random.randrange(0, T - t); spec_aug[:, :, t0:t0+t] = spec.mean()

def train_epoch_hybrid(model, dataloader, criterion, optimizer, device): 
    model.train(); running_loss = 0.0; correct_preds = 0; total_samples = 0; batch_num = 0
    for specs, stats, labels in dataloader: 
        batch_num += 1; preds = None 
        try:
            specs, stats, labels = specs.to(device), stats.to(device), labels.to(device); optimizer.zero_grad(); outputs = model(specs, stats); loss = criterion(outputs, labels)
            if loss is None or torch.isnan(loss): continue 
            preds = outputs.max(1)[1]; loss.backward(); optimizer.step() 
            running_loss += loss.item() * specs.size(0); total_samples += labels.size(0)
            if preds is not None: correct_preds += (preds == labels).sum().item() 
        except Exception as e: print(f"  Error (train batch {batch_num}): {e}"); continue 
    if total_samples == 0: return 0.0, 0.0; return running_loss / total_samples, correct_preds / total_samples

        X_spec_val_fold=all_specs_np[val_idx]; X_stat_val_fold_unscaled=all_stat_features_np[val_idx]; y_val_fold=all_labels_np[val_idx]

        scaler = StandardScaler(); X_stat_train_fold = scaler.fit_transform(X_stat_train_fold_unscaled); X_stat_val_fold = scaler.transform(X_stat_val_fold_unscaled)     

        train_dataset = HybridDataset(torch.from_numpy(X_spec_train_fold).float(), torch.from_numpy(X_stat_train_fold).float(), torch.from_numpy(y_train_fold).long(), spec_transform=current_train_transforms)
        val_dataset = HybridDataset(torch.from_numpy(X_spec_val_fold).float(), torch.from_numpy(X_stat_val_fold).float(), torch.from_numpy(y_val_fold).long(), spec_transform=val_test_transforms)
        train_loader = DataLoader(train_dataset, batch_size=BATCH_SIZE_OPT, shuffle=True, num_workers=0); val_loader = DataLoader(val_dataset, batch_size=BATCH_SIZE_OPT, shuffle=False, num_workers=0)

        model = HybridGeoModel(model_name=MODEL_NAME_OPT, pretrained=True, num_classes=NUM_CLASSES, num_stat_features=NUM_STAT_FEATURES, dropout_p=dropout_p, stat_feature_dim=stat_feature_dim, combined_dim=combined_dim)
        criterion = nn.CrossEntropyLoss(); params_to_train = [p for p in model.parameters() if p.requires_grad]; optimizer = optim.AdamW(params_to_train, lr=lr, weight_decay=weight_decay) 
        
        best_val_acc_fold = 0.0; best_val_loss_fold = float('inf'); current_best_fold_model_state = None; epochs_no_improve = 0; best_epoch = -1
        history_current_fold_obj = {'train_loss': [], 'train_acc': [], 'val_loss': [], 'val_acc': []}; fold_val_labels_final = []; fold_val_preds_final = [] 
        
        for epoch in range(1, EPOCHS_OPT + 1):
             train_loss, train_acc = train_epoch_hybrid(model, train_loader, criterion, optimizer, device)
             val_loss, val_acc, fold_val_labels_epoch, fold_val_preds_epoch = evaluate_model_hybrid(model, val_loader, criterion, device) 
             history_current_fold_obj['train_loss'].append(train_loss); history_current_fold_obj['train_acc'].append(train_acc); history_current_fold_obj['val_loss'].append(val_loss); history_current_fold_obj['val_acc'].append(val_acc)
             if val_loss < best_val_loss_fold: 
                 best_val_loss_fold = val_loss; best_val_acc_at_best_loss = val_acc; current_best_fold_model_state = deepcopy(model.state_dict()); epochs_no_improve = 0; best_epoch = epoch
                 fold_val_labels_final = fold_val_labels_epoch; fold_val_preds_final = fold_val_preds_epoch
             else: epochs_no_improve += 1
             if epochs_no_improve >= PATIENCE_OPT: break 
        
        current_fold_new_data_recall = 0.0; fold_new_data_labels=None; fold_new_data_preds=None 
        if current_best_fold_model_state and new_specs_np.size > 0:
             if MIN_VAL_ACC_THRESHOLD <= best_val_acc_at_best_loss <= MAX_VAL_ACC_THRESHOLD: 
                 model.load_state_dict(current_best_fold_model_state)
                 if new_stat_features_np.size > 0:
                     X_stat_new_scaled = scaler.transform(new_stat_features_np) # <<< שימוש ב-scaler מהאימון
                     new_dataset = HybridDataset(torch.from_numpy(new_specs_np).float(), torch.from_numpy(X_stat_new_scaled).float(), torch.from_numpy(new_labels_np).long(), spec_transform=val_test_transforms) 
                     new_loader = DataLoader(new_dataset, batch_size=BATCH_SIZE_OPT, shuffle=False, num_workers=0)
                     try:
                         _, _, fold_new_data_labels, fold_new_data_preds = evaluate_model_hybrid(model, new_loader, criterion, device) 
                         current_fold_new_data_recall = recall_score(fold_new_data_labels, fold_new_data_preds, labels=[LABEL_MAN], average='micro', zero_division=0)
                         print(f"    Fold {fold+1}: BestValLoss={best_val_loss_fold:.4f}(Acc={best_val_acc_at_best_loss:.4f}), NewRecall={current_fold_new_data_recall:.4f} (Ep {best_epoch}) -> QUALIFIED")
                         fold_best_models.append({ 'fold': fold+1, 'val_acc': best_val_acc_at_best_loss, 'new_recall': current_fold_new_data_recall, 'epoch': best_epoch, 'state_dict': current_best_fold_model_state, 'history': history_current_fold_obj, 'scaler': scaler, 
                                                  'val_labels': fold_val_labels_final, 'val_preds': fold_val_preds_final, 'new_labels': fold_new_data_labels, 'new_preds': fold_new_data_preds })
                     except Exception as eval_e: current_fold_new_data_recall = 0.0; print(f"    Fold {fold+1}: ValAcc={best_val_acc_at_best_loss:.4f}, NewRecall=EVAL FAILED: {eval_e}")
                 else: print(f"    Fold {fold+1}: Cannot eval new data - missing scaled stat features.")
             else: print(f"    Fold {fold+1}: ValAcc at best loss ({best_val_acc_at_best_loss:.4f}) is Outside threshold. Skip.")
        fold_recalls_new_data.append(current_fold_new_data_recall); fold_val_accs.append(best_val_acc_at_best_loss if best_epoch != -1 else 0) 
        try: trial.report(current_fold_new_data_recall, fold); If trial.should_prune(): print("  Trial pruned."); raise optuna.exceptions.TrialPruned()
        except Exception as e_prune: print(f"  Optuna reporting/pruning error: {e_prune}")
            
    average_recall = np.mean(fold_recalls_new_data) if fold_recalls_new_data else 0.0
    average_val_acc = np.mean(fold_val_accs) if fold_val_accs else 0.0
    print(f"--- Finished Trial {trial.number} (Avg ValAccAtBestLoss: {average_val_acc:.4f}, Avg NewRecall: {average_recall:.4f}, Time: {time.time() - start_trial_time:.1f}s) ---")
    # <<< ציון משולב לאופטימיזציה >>>
    combined_score = 0.4 * average_val_acc + 0.6 * average_recall # התאמת משקלים אולי?
    if fold_best_models: # שמור מידע רק אם היו מודלים מוצלחים
        trial_info = { 'trial_num': trial.number, 'combined_score': combined_score, 'avg_recall': average_recall, 'avg_val_acc': average_val_acc, 'params': trial.params, 'fold_models': fold_best_models }
        try: 
          current_best_score = study.user_attrs.get("best_overall_score", -1.0)
          if combined_score > current_best_score:
               study.set_user_attr("best_overall_score", combined_score)
               study.set_user_attr("best_trial_info", trial_info); print(f"  Trial {trial.number} is currently the best overall (Score: {combined_score:.4f}).")
        except Exception as e_attr: print(f"Could not set user_attr: {e_attr}")
    return combined_score # החזר את הציון המשולב
