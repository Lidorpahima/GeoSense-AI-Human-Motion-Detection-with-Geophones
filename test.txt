# (ודא שאתה מחליף את *כל* ההגדרה של המחלקה SpecAugment בקוד שלך בזו)

class SpecAugment(nn.Module):
    def __init__(self, freq_mask_param=20, time_mask_param=40, num_freq_masks=1, num_time_masks=1):
        super().__init__()
        self.freq_mask_param = freq_mask_param
        self.time_mask_param = time_mask_param
        self.num_freq_masks = num_freq_masks
        self.num_time_masks = num_time_masks

    def forward(self, spec):
        if not isinstance(spec, torch.Tensor):
             raise TypeError("Input must be a Tensor.")

        # --- >>> התיקון מתחיל כאן <<< ---
        # 1. בדוק אם הקלט הוא 3 ממדי
        if spec.ndim != 3:
            print(f"ERROR in SpecAugment: Expected 3D tensor (C, H, W), but got {spec.ndim}D tensor with shape {spec.shape}")
            # אפשר להחזיר את הקלט המקורי או לזרוק שגיאה
            print("Skipping SpecAugment due to unexpected input shape.")
            return spec # החזר את המקורי כדי למנוع קריסה

        spec_aug = spec.clone()
        # 2. השתמש בשמות ברורים: C=Channels, H=Height(Frequency), W=Width(Time)
        C, H, W = spec_aug.shape

        # 3. חשב פרמטרים עם הממדים הנכונים (H לתדר, W לזמן)
        # וודא שהממדים לא אפס לפני חלוקה
        if H <= 0 or W <= 0:
             print(f"Warning in SpecAugment: Input tensor has zero size dimension ({spec.shape}). Skipping.")
             return spec

        fm_param = min(self.freq_mask_param, H // 2) # fm_param תלוי ב-H (גובה/תדר)
        tm_param = min(self.time_mask_param, W // 2) # tm_param תלוי ב-W (רוחב/זמן)
        # --- >>> התיקון ממשיך בהמשך <<< ---

        mask_value = spec.mean() # חשב ערך למסכה

        # מסכת תדר (Frequency masking - פועלת על ממד הגובה H)
        for _ in range(self.num_freq_masks):
            if fm_param > 0: # ודא שגודל המסכה המקסימלי חיובי
                # שנה את חישוב f כדי שיכלול 0 ויגיע עד fm_param
                f = random.randrange(0, fm_param + 1) # גודל המסכה בין 0 ל-fm_param
                if H >= f and f > 0: # ודא שממד התדר גדול מגודל המסכה (והמסכה לא 0)
                    # שנה את חישוב f0 כדי שיגיע עד סוף הממד
                    f0 = random.randrange(0, H - f + 1) # אינדקס התחלה אפשרי
                    spec_aug[:, f0 : f0 + f, :] = mask_value # החל מסכה על כל הערוצים וכל הזמן

        # מסכת זמן (Time masking - פועלת על ממד הרוחב W)
        for _ in range(self.num_time_masks):
             if tm_param > 0: # ודא שגודל המסכה המקסימלי חיובי
                # שנה את חישוב t
                t = random.randrange(0, tm_param + 1) # גודל המסכה בין 0 ל-tm_param
                if W >= t and t > 0: # ודא שממד הזמן גדול מגודל המסכה (והמסכה לא 0)
                    # שנה את חישוב t0
                    t0 = random.randrange(0, W - t + 1) # אינדקס התחלה אפשרי
                    spec_aug[:, :, t0 : t0 + t] = mask_value # החל מסכה על כל הערוצים וכל התדרים

        return spec_aug
