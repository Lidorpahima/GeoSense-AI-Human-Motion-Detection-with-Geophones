# (פונקציות pad_channels, create_windows, create_mel_spectrogram, load_data_specs_signals_and_groups - ללא שינוי מהגרסה הקודמת)
def pad_channels(data, target_channels=max_channels):
    if data.ndim == 1: data = data.reshape(-1, 1)
    if data.shape[1] == target_channels: return data
    elif data.shape[1] < target_channels:
        padding = np.zeros((data.shape[0], target_channels - data.shape[1]), dtype=data.dtype)
        return np.hstack((data, padding))
    else:
        return data[:, :target_channels]

def create_windows(signal, label, window_size=window_size, stride=stride):
    X_windows, y_labels = [], []
    if signal.shape[0] < window_size: return [], []
    for start in range(0, signal.shape[0] - window_size + 1, stride):
        X_windows.append(signal[start : start + window_size, :])
        y_labels.append(label)
    return X_windows, y_labels

def create_mel_spectrogram(window_1d, sr=int(fs), n_fft=N_FFT, hop_length=HOP_LENGTH, n_mels=N_MELS, fmin=lowcut, fmax=highcut):
    try:
        signal = window_1d.astype(np.float32)
        if np.all(signal == signal[0]): return None
        effective_fmax = min(fmax, sr / 2 - 1)
        if effective_fmax <= fmin: effective_fmax = fmin + 10
        mel_spec = librosa.feature.melspectrogram(y=signal, sr=sr, n_fft=n_fft, hop_length=hop_length, n_mels=n_mels, fmin=fmin, fmax=effective_fmax)
        mel_spec_db = librosa.power_to_db(mel_spec, ref=np.max)
        min_db, max_db = mel_spec_db.min(), mel_spec_db.max()
        if max_db <= min_db: return None
        mel_spec_norm = (mel_spec_db - min_db) / (max_db - min_db) if max_db > min_db else np.zeros_like(mel_spec_db)
        if not np.isfinite(mel_spec_norm).all(): return None
        return mel_spec_norm.astype(np.float32)
    except Exception as e: return None

def load_data_specs_signals_and_groups(file_map, target_img_size=(IMG_HEIGHT_FINAL, IMG_WIDTH_FINAL)):
    all_specs, all_labels, all_signals, all_groups = [], [], [], []
    try: b, a = butter(order, [lowcut/(fs/2), highcut/(fs/2)], btype='band')
    except ValueError as e: print(f"CRITICAL FILTER ERROR: {e}. Aborting."); return [], [], [], []
    print(f"Loading data, signals, and groups...")
    items_created_count = 0; processed_files = 0
    group_counter = 0
    for label, info in file_map.items():
        filepaths = info.get('path', []); extra_args = info.get('args', {})
        if isinstance(filepaths, str): filepaths = [filepaths]
        if not filepaths: continue
        print(f"\nProcessing Label {label} ({len(filepaths)} files)...")
        for filepath in filepaths:
            processed_files += 1; group_id = group_counter
            print(f"  Processing file: {os.path.basename(filepath)} (Group ID: {group_id})...", end='')
            if not os.path.exists(filepath): print(f"    ERROR: File not found! Skipping."); group_counter += 1; continue
            df = None; data = None; windows_created_count = 0
            try:
                try: df = pd.read_csv(filepath, header=None, skipinitialspace=True, **extra_args)
                except Exception:
                     try: df = pd.read_csv(filepath, header=None, skiprows=1, skipinitialspace=True, **extra_args)
                     except Exception as e_read: print(f"    ERROR reading CSV {filepath}: {e_read}. Skipping."); group_counter += 1; continue
                if df is None or df.empty: print("    Empty file/read failed. Skipping."); group_counter += 1; continue
                df = df.dropna();
                if df.empty: print("    Empty after dropna. Skipping."); group_counter += 1; continue;
                try:
                   data = df.values
                   if not np.issubdtype(data.dtype, np.number): data = data.astype(float)
                except ValueError as e_type: print(f"    ERROR converting to numeric in {filepath}: {e_type}. Skipping."); group_counter += 1; continue
                if data.shape[0] < window_size or data.ndim < 2 or data.shape[1] == 0: print(f"    Insufficient data shape {data.shape}. Skipping."); group_counter += 1; continue
                data_padded = pad_channels(data)
                try: filtered_data = filtfilt(b, a, data_padded, axis=0)
                except Exception as e_filter: print(f"    ERROR filtering data in {filepath}: {e_filter}. Skipping file."); group_counter += 1; continue
                windows, labels_w = create_windows(filtered_data, label)
                if not windows: print("    No windows created. Skipping."); group_counter += 1; continue
                windows_created_count = len(windows); window_processed_count = 0
                for i, w in enumerate(windows):
                    try:
                        signal_to_process = w[:, 0].astype(np.float32)
                        mel_spec_norm = create_mel_spectrogram(signal_to_process)
                        if mel_spec_norm is None: continue
                        resized_spec = cv2.resize(mel_spec_norm, (target_img_size[1], target_img_size[0]), interpolation=cv2.INTER_LINEAR)
                        spec_3_channels = np.stack([resized_spec]*3, axis=0)
                        if not np.isfinite(spec_3_channels).all(): continue
                        all_signals.append(w.astype(np.float32))
                        all_specs.append(spec_3_channels)
                        all_labels.append(labels_w[i])
                        all_groups.append(group_id)
                        items_created_count += 1; window_processed_count += 1
                    except Exception as e_window: pass # print(f" Error window {i}: {e_window}")
                print(f" -> Processed {window_processed_count}/{windows_created_count} windows.")
            except Exception as e_file: print(f"    UNEXPECTED ERROR processing file {filepath}: {e_file}")
            group_counter += 1
    print(f"\nFinished loading. Processed {processed_files} files. Total items created: {items_created_count}")
    if not all_specs: print("WARNING: No items created.")
    specs_np = np.array(all_specs, dtype=np.float32) if all_specs else np.array([])
    labels_np = np.array(all_labels, dtype=np.int64) if all_labels else np.array([], dtype=np.int64)
    signals_np = np.array(all_signals, dtype=np.float32) if all_signals else np.array([])
    groups_np = np.array(all_groups, dtype=np.int64) if all_groups else np.array([], dtype=np.int64)
    del all_specs, all_labels, all_signals, all_groups
    return specs_np, labels_np, signals_np, groups_np

class SpecAugment(nn.Module):
    def __init__(self, freq_mask_param=20, time_mask_param=40, num_freq_masks=1, num_time_masks=1):
        super().__init__(); self.freq_mask_param = freq_mask_param; self.time_mask_param = time_mask_param
        self.num_freq_masks = num_freq_masks; self.num_time_masks = num_time_masks
    def forward(self, spec):
        if not isinstance(spec, torch.Tensor): raise TypeError("Input must be a Tensor."); spec_aug = spec.clone(); _, F, T = spec_aug.shape
        fm_param = min(self.freq_mask_param, F // 2); tm_param = min(self.time_mask_param, T // 2); mask_value = spec.mean()
        for _ in range(self.num_freq_masks):
            if fm_param > 0: f = random.randrange(0, fm_param);
            if F > f: f0 = random.randrange(0, F - f); spec_aug[:, f0:f0+f, :] = mask_value
        for _ in range(self.num_time_masks):
             if tm_param > 0: t = random.randrange(0, tm_param);
             if T > t: t0 = random.randrange(0, T - t); spec_aug[:, :, t0:t0+t] = mask_value
        return spec_aug

class SpectrogramDataset(Dataset):
    def __init__(self, specs_np, labels_np, transform=None):
        self.specs = torch.from_numpy(specs_np.astype(np.float32)); self.labels = torch.from_numpy(labels_np.astype(np.int64)); self.transform = transform
    def __len__(self): return len(self.labels)
    def __getitem__(self, idx):
        spec = self.specs[idx]; label = self.labels[idx]
        if self.transform: spec = self.transform(spec)
        return spec, label

def setup_efficientnet_model(model_name='efficientnet_b0', pretrained=True, num_classes=NUM_CLASSES, dropout_p=0.2):
    weights = EfficientNet_B0_Weights.DEFAULT if pretrained else None; model_arch = efficientnet_b0(weights=weights)
    num_ftrs = model_arch.classifier[1].in_features; model_arch.classifier = nn.Sequential(nn.Dropout(p=dropout_p, inplace=True), nn.Linear(num_ftrs, num_classes))
    if pretrained:
        for param in model_arch.features.parameters(): param.requires_grad = False
        for param in model_arch.classifier.parameters(): param.requires_grad = True
    model_arch = model_arch.to(DEVICE); return model_arch

def train_epoch_effnet(model, dataloader, criterion, optimizer, device):
    model.train(); running_loss = 0.0; correct_preds = 0; total_samples = 0;
    for inputs, labels in dataloader:
        try:
            inputs, labels = inputs.to(device), labels.to(device); optimizer.zero_grad(); outputs = model(inputs); loss = criterion(outputs, labels)
            if loss is None or torch.isnan(loss): print("  Warning: NaN/None loss in train. Skipping batch."); continue
            preds = outputs.max(1)[1]; loss.backward(); optimizer.step()
            running_loss += loss.item() * inputs.size(0); correct_preds += (preds == labels).sum().item(); total_samples += labels.size(0)
        except Exception as e: print(f"  Error train batch: {e}"); continue
    if total_samples == 0: return 0.0, 0.0; return running_loss / total_samples, correct_preds / total_samples

def evaluate_model_effnet(model, dataloader, criterion, device):
    model.eval(); running_loss = 0.0; correct_preds = 0; total_samples = 0; all_preds = []; all_labels = []
    with torch.no_grad():
        for inputs, labels in dataloader:
            try:
                inputs, labels = inputs.to(device), labels.to(device); outputs = model(inputs); loss = criterion(outputs, labels);
                if loss is None or torch.isnan(loss): print("  Warning: NaN/None loss in eval. Skipping batch."); continue
                preds = outputs.max(1)[1]
                running_loss += loss.item() * inputs.size(0); correct_preds += (preds == labels).sum().item(); total_samples += labels.size(0)
                all_preds.extend(preds.cpu().numpy()); all_labels.extend(labels.cpu().numpy())
            except Exception as e: print(f"  Error eval batch: {e}"); continue
    if total_samples == 0: return 0.0, 0.0, [], []; return running_loss / total_samples, correct_preds / total_samples, all_labels, all_preds

print("\n--- Defining Optuna Objective Function ---")

def objective(trial):
    start_trial_time = time.time()
    print(f"\n{'='*15} Optuna Trial {trial.number} Starting {'='*15}")

    # --- דגימת היפרפרמטרים ---
    lr = trial.suggest_float("lr", 1e-4, 5e-3, log=True)
    dropout_p = trial.suggest_float("dropout_p", 0.2, 0.5, step=0.1)
    weight_decay = trial.suggest_float("weight_decay", 1e-5, 1e-3, log=True)
    specaug_freq = trial.suggest_int("specaug_freq", 5, 25)
    specaug_time = trial.suggest_int("specaug_time", 10, 50)
    specaug_f_masks = trial.suggest_int("specaug_f_masks", 1, 2)
    specaug_t_masks = trial.suggest_int("specaug_t_masks", 1, 2)
    print(f"  Trial Params: LR={lr:.1e}, Dropout={dropout_p:.2f}, WD={weight_decay:.1e}, SpecAugF={specaug_freq}, SpecAugT={specaug_time}, N_F={specaug_f_masks}, N_T={specaug_t_masks}")

    current_train_transforms = transforms.Compose([
        SpecAugment(freq_mask_param=specaug_freq, time_mask_param=specaug_time, num_freq_masks=specaug_f_masks, num_time_masks=specaug_t_masks),
        transforms.Normalize(mean=imagenet_mean, std=imagenet_std) ])

    kfold = StratifiedGroupKFold(n_splits=N_SPLITS_OPT, shuffle=True, random_state=trial.number + SEED_OFFSET)
    fold_val_accs = []; fold_new_data_accs = []
    qualified_fold_results_in_trial = [] # כולל state_dict אם עמד בתנאי הולידציה הראשוני

    X, y, groups = orig_specs_np, orig_labels_np, orig_groups_np
    unique_groups = np.unique(groups)
    if len(unique_groups) < N_SPLITS_OPT:
        print(f"  WARNING: Unique groups ({len(unique_groups)}) < N_SPLITS_OPT ({N_SPLITS_OPT}). StratifiedGroupKFold might fail or behave unexpectedly.")
        # Consider raising TrialPruned here if this is critical

    fold_iterator = kfold.split(X, y, groups)

    for fold, (train_ids, val_ids) in enumerate(fold_iterator):
        fold_start_time = time.time()
        print(f"  --- Fold {fold+1}/{N_SPLITS_OPT} ---")
        train_groups, val_groups = np.unique(groups[train_ids]), np.unique(groups[val_ids]); overlap = len(set(train_groups) & set(val_groups))
        print(f"    Train groups: {len(train_groups)}, Val groups: {len(val_groups)}, Overlap: {overlap}")
        if overlap > 0: print("    !!!! CRITICAL WARNING: Group overlap detected between train/val sets. Leakage occurring! !!!!")

        X_train_fold, y_train_fold = X[train_ids], y[train_ids]
        X_val_fold, y_val_fold = X[val_ids], y[val_ids]
        if len(X_train_fold) == 0 or len(X_val_fold) == 0:
            print(f"    Warning: Fold {fold+1} has empty train ({len(X_train_fold)}) or validation ({len(X_val_fold)}) set. Skipping fold."); fold_val_accs.append(0.0); fold_new_data_accs.append(0.0); continue

        train_dataset = SpectrogramDataset(X_train_fold, y_train_fold, transform=current_train_transforms)
        val_dataset = SpectrogramDataset(X_val_fold, y_val_fold, transform=val_test_transforms)
        use_pin_memory = (DEVICE == torch.device("cuda"))
        train_loader = DataLoader(train_dataset, batch_size=BATCH_SIZE_OPT, shuffle=True, num_workers=0, pin_memory=use_pin_memory)
        val_loader = DataLoader(val_dataset, batch_size=BATCH_SIZE_OPT, shuffle=False, num_workers=0, pin_memory=use_pin_memory)

        model = setup_efficientnet_model(model_name=MODEL_NAME_OPT, pretrained=True, num_classes=NUM_CLASSES, dropout_p=dropout_p)
        criterion = nn.CrossEntropyLoss()
        params_to_optimize = model.classifier.parameters() if not any(p.requires_grad for p in model.features.parameters()) else model.parameters()
        optimizer = optim.AdamW(params_to_optimize, lr=lr, weight_decay=weight_decay)

        best_val_acc_fold = 0.0; current_best_fold_model_state = None; epochs_no_improve = 0; best_epoch = -1
        history_current_fold = {'train_loss': [], 'train_acc': [], 'val_loss': [], 'val_acc': []}
        best_fold_val_labels = []; best_fold_val_preds = []
        fold_new_data_labels = []; fold_new_data_preds = []
        current_fold_new_data_acc = 0.0

        print(f"    Starting training for {EPOCHS_OPT} epochs...")
        for epoch in range(1, EPOCHS_OPT + 1):
            epoch_start_time = time.time()
            train_loss, train_acc = train_epoch_effnet(model, train_loader, criterion, optimizer, device)
            val_loss, val_acc, fold_val_labels_epoch, fold_val_preds_epoch = evaluate_model_effnet(model, val_loader, criterion, device)
            epoch_time = time.time() - epoch_start_time
            print(f"    Epoch {epoch}/{EPOCHS_OPT} | Train Loss: {train_loss:.4f}, Acc: {train_acc:.4f} | Val Loss: {val_loss:.4f}, Acc: {val_acc:.4f} | Time: {epoch_time:.1f}s")
            history_current_fold['train_loss'].append(train_loss); history_current_fold['train_acc'].append(train_acc); history_current_fold['val_loss'].append(val_loss); history_current_fold['val_acc'].append(val_acc)
            if val_acc > best_val_acc_fold:
                best_val_acc_fold = val_acc; current_best_fold_model_state = deepcopy(model.state_dict()); epochs_no_improve = 0; best_epoch = epoch
                best_fold_val_labels = fold_val_labels_epoch; best_fold_val_preds = fold_val_preds_epoch
                # print(f"      New best val_acc: {best_val_acc_fold:.4f} at epoch {epoch}.") # Optional: uncomment for more verbose logging
            else: epochs_no_improve += 1
            if epochs_no_improve >= PATIENCE_OPT: print(f"    Early stopping at epoch {epoch}. Best val_acc: {best_val_acc_fold:.4f} at epoch {best_epoch}."); break

        # --- הערכה על נתונים חדשים + שמירה מותנית ---
        if current_best_fold_model_state and new_specs_np.size > 0:
            print(f"\n    Fold {fold+1} finished. Best Val Acc (Epoch {best_epoch}): {best_val_acc_fold:.4f}.")
            if MIN_VAL_ACC_THRESHOLD <= best_val_acc_fold <= MAX_VAL_ACC_THRESHOLD:
                print(f"      Val Acc within threshold [{MIN_VAL_ACC_THRESHOLD}, {MAX_VAL_ACC_THRESHOLD}]. Evaluating on new data...")
                model.load_state_dict(current_best_fold_model_state) # Load best state for this fold
                new_dataset = SpectrogramDataset(new_specs_np, new_labels_np, transform=val_test_transforms)
                new_loader = DataLoader(new_dataset, batch_size=BATCH_SIZE_OPT, shuffle=False, num_workers=0, pin_memory=use_pin_memory)
                try:
                    new_loss, new_acc, fold_new_data_labels_eval, fold_new_data_preds_eval = evaluate_model_effnet(model, new_loader, criterion, device)
                    current_fold_new_data_acc = new_acc
                    fold_new_data_labels = fold_new_data_labels_eval
                    fold_new_data_preds = fold_new_data_preds_eval
                    print(f"        -> New Data Eval: Loss={new_loss:.4f}, Accuracy={current_fold_new_data_acc:.4f}")

                    # <<< שמור מידע על ה-fold (כולל state_dict) אם עמד בתנאי הולידציה הראשוני >>>
                    # This dict is used later to find the best fold *within* this trial
                    qualified_fold_results_in_trial.append({
                        'fold': fold + 1, 'val_acc': best_val_acc_fold, 'new_acc': current_fold_new_data_acc,
                        'epoch': best_epoch, 'history': history_current_fold,
                        'val_labels': best_fold_val_labels, 'val_preds': best_fold_val_preds,
                        'new_labels': fold_new_data_labels, 'new_preds': fold_new_data_preds,
                        'state_dict': deepcopy(current_best_fold_model_state) # Include state_dict here!
                    })

                    # <<< --- בדיקה ושמירה מותנית נוספת --- >>>
                    if best_val_acc_fold >= SAVE_THRESHOLD_VAL and current_fold_new_data_acc >= SAVE_THRESHOLD_NEW:
                        print(f"    **** Fold {fold+1} (Trial {trial.number}) meets saving criteria (Val: {best_val_acc_fold:.4f} >= {SAVE_THRESHOLD_VAL}, New: {current_fold_new_data_acc:.4f} >= {SAVE_THRESHOLD_NEW})! Saving... ****")
                        try:
                            model_filename_cond = os.path.join(QUALIFIED_MODELS_DIR,
                                f"model_trial_{trial.number}_fold_{fold+1}_val_{best_val_acc_fold:.3f}_new_{current_fold_new_data_acc:.3f}.pth")
                            save_data_fold = {
                                'model_state_dict': current_best_fold_model_state, 'trial_number': trial.number, 'fold': fold + 1,
                                'epoch': best_epoch, 'val_acc': best_val_acc_fold, 'test_acc_normalized': current_fold_new_data_acc,
                                'model_name': MODEL_NAME_OPT, 'hyperparameters': trial.params, 'imagenet_mean': imagenet_mean,
                                'imagenet_std': imagenet_std, 'class_names': CLASS_NAMES, 'N_MELS': N_MELS, 'N_FFT': N_FFT,
                                'HOP_LENGTH': HOP_LENGTH, 'IMG_HEIGHT_FINAL': IMG_HEIGHT_FINAL, 'IMG_WIDTH_FINAL': IMG_WIDTH_FINAL,
                                'window_size': window_size, 'stride': stride, 'fs': fs, 'lowcut': lowcut, 'highcut': highcut, 'order': order,
                                'kfold_random_state': trial.number + SEED_OFFSET
                            }
                            torch.save(save_data_fold, model_filename_cond)
                            print(f"      Saved qualified model to: {model_filename_cond}")
                            summary_filename_cond = model_filename_cond.replace(".pth", "_summary.txt")
                            with open(summary_filename_cond, 'w', encoding='utf-8') as f_sum:
                               f_sum.write(f"Qualified Model Summary\nTrial: {trial.number}, Fold: {fold+1}, Best Epoch: {best_epoch}\nVal Acc: {best_val_acc_fold:.4f}\nNew Data Acc: {current_fold_new_data_acc:.4f}\n\nHyperparameters:\n");
                               for k,v in trial.params.items(): f_sum.write(f"  {k}: {v}\n")
                            print(f"      Saved qualified summary to: {summary_filename_cond}")
                        except Exception as e_save_cond: print(f"      ERROR saving qualified model: {e_save_cond}")
                    # <<< --- סוף שמירה מותנית --- >>>

                except Exception as e_eval_new: print(f"        ERROR during new data evaluation: {e_eval_new}"); current_fold_new_data_acc = 0.0 # Reset acc if eval failed
            else: print(f"      Val Acc ({best_val_acc_fold:.4f}) outside threshold [{MIN_VAL_ACC_THRESHOLD}, {MAX_VAL_ACC_THRESHOLD}]. Skipping new data evaluation.")
        elif new_specs_np.size == 0: print(f"    No new data loaded. Skipping new data evaluation.")
        else: print(f"    No best model state found for fold {fold+1}. Skipping new data evaluation.")

        fold_val_accs.append(best_val_acc_fold)
        fold_new_data_accs.append(current_fold_new_data_acc) # Will be 0 if not evaluated

        trial.report(best_val_acc_fold, fold) # Report intermediate val_acc for pruning
        if trial.should_prune(): print(f"  Trial {trial.number} pruned at fold {fold+1}."); raise optuna.exceptions.TrialPruned()
        print(f"  --- Fold {fold+1} Finished. Time: {time.time() - fold_start_time:.1f}s ---")
        del model, optimizer, criterion, train_loader, val_loader, train_dataset, val_dataset
        if DEVICE == torch.device("cuda"): torch.cuda.empty_cache()

    # --- סוף ה-Trial ---
    average_val_acc = np.mean(fold_val_accs) if fold_val_accs else 0.0
    qualified_new_accs = [f['new_acc'] for f in qualified_fold_results_in_trial] # Accs from folds that ran new data eval
    average_acc_new_data = np.mean(qualified_new_accs) if qualified_new_accs else 0.0 # Optuna objective

    print(f"\n--- Finished Trial {trial.number} ---")
    print(f"  Avg Val Acc (Leak-Free, all folds): {average_val_acc:.4f}")
    print(f"  Avg Test Acc (Normalized Data, qualified folds only): {average_acc_new_data:.4f}")
    print(f"  Number of qualified folds (for new data eval): {len(qualified_new_accs)}/{N_SPLITS_OPT}")
    print(f"  Total Trial Time: {time.time() - start_trial_time:.1f}s")
    print(f"{'='*40}")

    # --- שמירת מידע ו-state_dict הטוב ביותר מה-Trial ב-User Attributes ---
    trial_summary = { 'trial_num': trial.number, 'params': trial.params, 'avg_val_acc': average_val_acc, 'avg_test_acc_new': average_acc_new_data,
        'qualified_folds_metrics': [{'fold': f['fold'], 'val_acc': f['val_acc'], 'new_acc': f['new_acc']} for f in qualified_fold_results_in_trial] }
    trial.set_user_attr("trial_summary", trial_summary)
    # Store results *without* state_dict for general info
    fold_results_no_state = [{k: v for k, v in fold_res.items() if k != 'state_dict'} for fold_res in qualified_fold_results_in_trial]
    trial.set_user_attr("qualified_fold_results_no_state", fold_results_no_state)

    best_fold_in_trial_state_dict = None; best_fold_num_in_trial = -1
    if qualified_fold_results_in_trial: # If there were folds that met the *initial* val threshold
        qualified_fold_results_in_trial.sort(key=lambda x: (-x['val_acc'], -x['new_acc'])) # Sort by Val then New acc
        best_qualified_fold_info = qualified_fold_results_in_trial[0]
        best_fold_in_trial_state_dict = best_qualified_fold_info['state_dict'] # Get state_dict from the best one
        best_fold_num_in_trial = best_qualified_fold_info['fold']
        print(f"  Identified best fold in trial: Fold {best_fold_num_in_trial} (ValAcc: {best_qualified_fold_info['val_acc']:.4f}, NewAcc: {best_qualified_fold_info['new_acc']:.4f})")
        # Store ONLY the state_dict of the single best fold for this trial
        trial.set_user_attr("best_fold_state_dict", best_fold_in_trial_state_dict)
        trial.set_user_attr("best_fold_num_in_trial", best_fold_num_in_trial)
        print(f"  Stored state_dict for best fold ({best_fold_num_in_trial}) in trial attributes.")
    else:
        print("  No qualified folds (meeting initial val threshold) in this trial to determine the best state_dict.")
        trial.set_user_attr("best_fold_state_dict", None)
        trial.set_user_attr("best_fold_num_in_trial", -1)

    # Return the main objective for Optuna
    return average_acc_new_data
